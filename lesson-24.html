🎓 أهلاً بعالم الأداء!

كل تطبيقاتنا حتى الآن كانت صحيحة وظيفياً، لكن هل هي "سريعة" فعلاً مع بيانات كبيرة؟

اليوم نتعلّم لماذا يُعيد React رسم مكوّنات لا داعي لإعادة رسمها، وثلاث أدوات رسمية لمنع ذلك — مع تحذير مهم: هذه الأدوات ليست "افتراضية"، ولها ثمن، فاستخدامها في غير محلّها قد يُبطئ تطبيقك بدل تسريعه!

🐌

الجزء الأول: لماذا تُعاد رسمة مكوّنات لا داعي لها؟

💡 القاعدة الأساسية

عندما تُعاد رسم مكوّن أب (Parent) نتيجة تغيّر حالته، يُعاد رسم كل أبنائه (Children) افتراضياً — حتى لو لم تتغيّر أي Prop يستقبلونها!

1.1

مثال المشكلة

JSX 
function App() {
  const [count, setCount] = useState(0);
  const [text, setText] = useState('');

  return (
    <div>
      <input value={text} onChange={(e) => setText(e.target.value)} />
      <button onClick={() => setCount(count + 1)}>{count}</button>
      <HeavyList /> {/* مكوّن "ثقيل" لا علاقة له بـ text أو count إطلاقاً */}
    </div>
  );
}

function HeavyList() {
  console.log('🐌 HeavyList يُعاد رسمه!');
  // تخيّل حسابات معقّدة أو قائمة بآلاف العناصر هنا...
  return <ul>{/* ... */}</ul>;
}
⚠️ لاحظ

كل حرف يُكتَب في حقل النص، وكل ضغطة على الزر، تُعيد رسم App بالكامل — وبالتبعية HeavyList أيضاً، رغم أنه لا يعتمد على text ولا count إطلاقاً!

🧊

الجزء الثاني: React.memo

2.1

الفكرة: تجميد المكوّن ما لم تتغيّر Props فعلياً

JSX 
import { memo } from 'react';

const HeavyList = memo(function HeavyList() {
  console.log('🐌 HeavyList يُعاد رسمه!');
  return <ul>{/* ... */}</ul>;
});

export default HeavyList;
// الآن: HeavyList لن يُعاد رسمه إلا إذا تغيّرت Props الخاصة به فعلياً
2.2

تحذير: المقارنة السطحية (Shallow Comparison)

JSX 
const UserCard = memo(function UserCard({ user }) {
  return <p>{user.name}</p>;
});

function App() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <button onClick={() => setCount(count + 1)}>{count}</button>
      {/* ❌ كائن جديد يُنشأ في كل رسم، رغم تطابق المحتوى منطقياً! */}
      <UserCard user={{ name: 'سارة' }} />
    </div>
  );
}
// { name: 'سارة' } كائن جديد بمرجع مختلف في كل مرة
// → memo يقارن المرجع فقط (Shallow) → يراها "تغيّرت" → يُعاد الرسم رغم تطابق البيانات!
⭐ القاعدة الذهبية

React.memo يقارن Props بمقارنة سطحية (===) — تنجح مع النصوص والأرقام، لكنها تفشل مع كائنات أو مصفوفات أو دوال جديدة تُنشأ في كل رسم.

🧮

الجزء الثالث: useMemo

3.1

المشكلة: حساب مكلف يتكرر بلا داعٍ

JSX 
function ProductList({ products, searchTerm }) {
  // ❌ هذا الفلتر يُعاد تنفيذه في كل رسم، حتى لو لم يتغيّر searchTerm أو products!
  const filtered = products.filter((p) => p.name.includes(searchTerm));

  return <ul>{filtered.map((p) => <li key={p.id}>{p.name}</li>)}</ul>;
}
3.2

useMemo كحل

JSX 
import { useMemo } from 'react';

function ProductList({ products, searchTerm }) {
  const filtered = useMemo(() => {
    console.log('🧮 إعادة حساب الفلترة...');
    return products.filter((p) => p.name.includes(searchTerm));
  }, [products, searchTerm]); // يُعاد الحساب فقط عند تغيّر أحدهما فعلياً

  return <ul>{filtered.map((p) => <li key={p.id}>{p.name}</li>)}</ul>;
}
🔗

الجزء الرابع: useCallback

4.1

المشكلة: دالة جديدة تكسر memo

JSX 
const ExpensiveButton = memo(function ExpensiveButton({ onClick }) {
  console.log('🐌 ExpensiveButton يُعاد رسمه!');
  return <button onClick={onClick}>زر مكلف</button>;
});

function App() {
  const [count, setCount] = useState(0);

  // ❌ دالة جديدة تُنشأ في كل رسم، حتى لو كان محتواها متطابقاً منطقياً
  function handleClick() {
    console.log('تم الضغط');
  }

  return (
    <div>
      <button onClick={() => setCount(count + 1)}>{count}</button>
      <ExpensiveButton onClick={handleClick} /> {/* memo هنا لن يمنع إعادة الرسم! */}
    </div>
  );
}
4.2

useCallback كحل

JSX 
import { useCallback } from 'react';

function App() {
  const [count, setCount] = useState(0);

  const handleClick = useCallback(() => {
    console.log('تم الضغط');
  }, []); // نفس مرجع الدالة يبقى ثابتاً بين كل الرسومات

  return (
    <div>
      <button onClick={() => setCount(count + 1)}>{count}</button>
      <ExpensiveButton onClick={handleClick} /> {/* الآن memo يعمل فعلاً! */}
    </div>
  );
}
💡 لماذا لم يكفِ memo وحده؟

onClick={handleClick} كانت تحمل دالة جديدة بمرجع مختلف في كل رسم، فرأى memo أن Props "تغيّرت" (المقارنة السطحية تقارن مراجع الدوال لا محتواها) فأعاد الرسم رغم عدم الحاجة. useCallback يُبقي نفس مرجع الدالة، فيعمل memo كما هو متوقَّع.

⚖️

الجزء الخامس: متى نستخدم هذه الأدوات فعلاً؟

السؤالالقرار
هل المكوّن بسيط وخفيف (نص، أيقونة صغيرة)؟❌ لا تستخدم memo — التكلفة أكبر من الفائدة
هل المكوّن "ثقيل" فعلاً (قائمة ضخمة، رسم بياني، حسابات معقّدة)؟✅ فكِّر بـ memo
هل تُمرَّر دالة كـ Prop لمكوّن memo؟✅ استخدم useCallback لتلك الدالة
هل الحساب فعلاً مكلف (فرز/فلترة آلاف العناصر)؟✅ استخدم useMemo
هل تُحسِّن "افتراضياً" بلا مشكلة أداء ملحوظة فعلياً؟⚠️ توقّف — تحسين سابق لأوانه
⭐ القاعدة الذهبية

قِس أولاً، ثم حسِّن (Measure First, Then Optimize). لكل من memo وuseMemo وuseCallback تكلفة حسابية بسيطة خاصة به — الإفراط في استخدامها في كل مكان قد يُبطئ تطبيقك فعلياً بدل تسريعه. ابدأ بلا أي تحسين، وأضفه فقط عند رصد مشكلة أداء حقيقية.

الجزء السادس: المشروع — قائمة ضخمة محسَّنة الأداء

قائمة منتجات بـ 1000 عنصر، مع بحث فوري وإضافة للسلة — بدون أي إعادة رسم غير ضرورية للعناصر الفردية.

src/components/ProductItem.jsx

JSX 
import { memo } from 'react';

const ProductItem = memo(function ProductItem({ product, onAddToCart }) {
  console.log(`🎨 رسم: ${product.name}`);
  return (
    <li className="product-item">
      <span>{product.name} — {product.price} $</span>
      <button onClick={() => onAddToCart(product.id)}>أضف للسلة</button>
    </li>
  );
});

export default ProductItem;

src/App.jsx

JSX 
import { useState, useMemo, useCallback } from 'react';
import ProductItem from './components/ProductItem';

const allProducts = Array.from({ length: 1000 }, (_, i) => ({
  id: i,
  name: `منتج ${i + 1}`,
  price: (i % 50) + 10,
}));

function App() {
  const [searchTerm, setSearchTerm] = useState('');
  const [cartCount, setCartCount] = useState(0);

  // useMemo: الفلترة الثقيلة (1000 عنصر) لا تُعاد إلا عند تغيّر searchTerm فعلياً
  const filteredProducts = useMemo(() => {
    return allProducts.filter((p) => p.name.includes(searchTerm));
  }, [searchTerm]);

  // useCallback: نفس مرجع الدالة يبقى ثابتاً، فلا يُكسَر memo كل ProductItem
  const handleAddToCart = useCallback((id) => {
    setCartCount((prev) => prev + 1);
  }, []);

  return (
    <div>
      <input
        value={searchTerm}
        onChange={(e) => setSearchTerm(e.target.value)}
        placeholder="ابحث عن منتج..."
      />
      <p>🛒 السلة: {cartCount}</p>

      <ul>
        {filteredProducts.map((product) => (
          <ProductItem key={product.id} product={product} onAddToCart={handleAddToCart} />
        ))}
      </ul>
    </div>
  );
}

export default App;
✅ النتيجة

الكتابة في حقل البحث تُعيد حساب الفلترة فقط (متوقَّع ومطلوب)، لكنها لا تُعيد رسم عناصر ProductItem الفردية التي لم تتغيّر — بفضل memo وuseCallback معاً.

  • أصلحه باستخدام React.memo وتأكد من اختفاء الرسائل غير الضرورية في الكونسول

✏️ تمرين 2 — useMemo عملي20 دقيقة

  • مصفوفة من 1000+ عنصر (أرقام أو منتجات)
  • دالة فرز حسب السعر أو القيمة، مُغلَّفة بـ useMemo
  • أثبت عبر console.log أن الفرز لا يُعاد إلا عند تغيّر معيار الفرز فعلياً

✏️ تمرين 3 — useCallback مع memo معاً25 دقيقة

  • قائمة عناصر قابلة للحذف، كل عنصر مكوّن memo مستقل
  • دالة الحذف مُغلَّفة بـ useCallback
  • تأكد أن حذف عنصر واحد لا يُعيد رسم بقية العناصر (console.log لكل عنصر)

الواجب: تحسين أداء تطبيق قائمة مهام

خُذ تطبيق "قائمة مهام" من جلسة سابقة (أو أنشئ نسخة جديدة بـ 500+ مهمة وهمية)، وحسِّن أداءه:

◆ المتطلبات:

  • كل عنصر مهمة مكوّن TaskItem منفصل مُغلَّف بـ React.memo
  • كل الدوال المُمرَّرة لـ TaskItem (حذف، تبديل حالة) مُغلَّفة بـ useCallback
  • فلترة/بحث في المهام مُغلَّف بـ useMemo
  • أثبت بـ console.log أن الكتابة في حقل الإضافة لا تُعيد رسم عناصر القائمة الموجودة

◆ متطلبات تقنية:

  • قارن عدد مرات الرسم قبل وبعد التحسين (بالكونسول) واكتب ملاحظاتك في README
  • لا تُفرِط في استخدام memo على مكوّنات بسيطة لا تستحق ذلك
