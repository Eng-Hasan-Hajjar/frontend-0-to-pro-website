الجزء الأول: جلب البيانات في Server Components

1.1

async/await مباشرة داخل المكوّن

JSX 
// app/products/page.js — Server Component، بلا useEffect إطلاقاً!
async function getProducts() {
  const res = await fetch('https://api.example.com/products');
  if (!res.ok) throw new Error('فشل جلب المنتجات');
  return res.json();
}

export default async function ProductsPage() {
  const products = await getProducts();

  return (
    <div>
      <h1>🛍️ المنتجات</h1>
      <ul>
        {products.map((p) => (
          <li key={p.id}>{p.name} — {p.price} $</li>
        ))}
      </ul>
    </div>
  );
}
💡 لاحظ الفرق

لا حاجة لـ useState أو useEffect أو حالات loading/error يدوية — الدالة نفسها async، وNext.js "ينتظر" اكتمالها على الخادم قبل إرسال الصفحة الجاهزة كاملة للمتصفح.

الجزء الثاني: التخزين المؤقت (Caching) والتحقّق منه

2.1

السلوك الافتراضي والتحكّم به

React 
// افتراضياً: Next.js يُخزِّن نتيجة fetch مؤقتاً (يشبه SSG لهذا الطلب بالذات)
const res = await fetch('https://api.example.com/products');

// لتعطيل التخزين المؤقت تماماً (بيانات حيّة في كل طلب، كـ SSR)
const res2 = await fetch('https://api.example.com/products', {
  cache: 'no-store',
});

// إعادة التحقّق (Revalidation) كل 60 ثانية بدل تخزين دائم أو تعطيل كامل
const res3 = await fetch('https://api.example.com/products', {
  next: { revalidate: 60 },
});
💡 الفكرة

هذا يمنحك تحكّماً دقيقاً: بيانات نادرة التغيير (شبيه بـ SSG)، بيانات حيّة تماماً مع كل طلب (شبيه بـ SSR)، أو نقطة وسط بإعادة تحقّق دورية (ISR).

إضافة معالجة POST لنفس الملف

React 
// app/api/products/route.js (بالإضافة إلى GET أعلاه)
export async function POST(request) {
  const body = await request.json(); // قراءة بيانات الطلب المُرسَلة

  const newProduct = { id: Date.now(), ...body };
  // هنا عادة نحفظ في قاعدة بيانات حقيقية — نكتفي بإعادة الكائن للتوضيح

  return Response.json(newProduct, { status: 201 });
}

الجزء الرابع: ربط Client Component بـ API Route داخلي

4.1

نموذج يُرسل بيانات لنقطة نهاية داخلية

JSX 
// app/components/AddProductForm.js
'use client';

import { useState } from 'react';

export default function AddProductForm() {
  const [name, setName] = useState('');
  const [status, setStatus] = useState('');

  async function handleSubmit(e) {
    e.preventDefault();
    setStatus('جارٍ الإرسال...');

    const res = await fetch('/api/products', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, price: 0 }),
    });

    if (res.ok) {
      setStatus('✅ تمت الإضافة بنجاح');
      setName('');
    } else {
      setStatus('❌ حدث خطأ');
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={name} onChange={(e) => setName(e.target.value)} placeholder="اسم المنتج" />
      <button type="submit">إضافة</button>
      <p>{status}</p>
    </form>
  );
}
💡 لاحظ

الرابط النسبي "/api/products" — لا حاجة لعنوان خادم كامل، لأن API Route يعيش داخل نفس مشروع Next.js تماماً.

الجزء الخامس: المشروع — تطبيق ملاحظات كامل

تطبيق ملاحظات بواجهة أمامية (Server Component يجلب البيانات) وAPI خلفي داخلي (GET وPOST) في نفس المشروع.

app/api/notes/route.js

React 
let notes = [
  { id: 1, text: 'أول ملاحظة' },
];

export async function GET() {
  return Response.json(notes);
}

export async function POST(request) {
  const body = await request.json();
  const newNote = { id: Date.now(), text: body.text };
  notes.push(newNote);
  return Response.json(newNote, { status: 201 });
}

app/notes/page.js

JSX 
import AddNoteForm from '../components/AddNoteForm';

async function getNotes() {
  const res = await fetch('http://localhost:3000/api/notes', { cache: 'no-store' });
  return res.json();
}

export default async function NotesPage() {
  const notes = await getNotes();

  return (
    <div>
      <h1>📝 الملاحظات</h1>
      <AddNoteForm />
      <ul>
        {notes.map((note) => (
          <li key={note.id}>{note.text}</li>
        ))}
      </ul>
    </div>
  );
}

app/components/AddNoteForm.js

JSX 
'use client';

import { useState } from 'react';
import { useRouter } from 'next/navigation';

export default function AddNoteForm() {
  const [text, setText] = useState('');
  const router = useRouter();

  async function handleSubmit(e) {
    e.preventDefault();
    await fetch('/api/notes', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text }),
    });
    setText('');
    router.refresh(); // يُعيد جلب بيانات Server Component دون فقدان حالة الصفحة
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={text} onChange={(e) => setText(e.target.value)} placeholder="ملاحظة جديدة..." />
      <button type="submit">إضافة</button>
    </form>
  );
}
💡 router.refresh()

من next/navigation — يُعيد جلب بيانات صفحة NotesPage (وهي Server Component) دون إعادة تحميل كامل للمتصفح أو فقدان أي حالة محلية أخرى في الصفحة.

الواجب: لوحة تعليقات (Comments Board)

أنشئ تطبيق Next.js للوحة تعليقات، بواجهة أمامية وAPI خلفي داخلي كامل:

◆ المتطلبات الوظيفية:

  • صفحة رئيسية Server Component تجلب كل التعليقات عبر GET من /api/comments
  • نموذج Client Component لإضافة تعليق جديد (اسم + نص) عبر POST لنفس المسار
  • تحديث القائمة فوراً بعد الإضافة عبر router.refresh()
  • عرض عدد التعليقات الإجمالي أعلى الصفحة

◆ متطلبات تقنية:

  • ملف واحد app/api/comments/route.js يحوي كلاً من GET وPOST
  • التحقّق من عدم إرسال نص تعليق فارغ (في الواجهة وفي API Route معاً)
  • استخدام cache: "no-store" عند جلب التعليقات لضمان ظهورها فوراً بعد الإضافة
