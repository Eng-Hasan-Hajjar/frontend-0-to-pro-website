الجزء الأول: طرق المصفوفات عالية الترتيب

الطرق عالية الترتيب (Higher-Order Methods) تأخذ دالة كمعامل — هي أساس البرمجة الوظيفية في JavaScript وتُغني عن كثير من الحلقات.

1.1

map() — تحويل كل عنصر

map() تُعيد مصفوفة جديدة بنفس الحجم، كل عنصر فيها هو نتيجة تطبيق الدالة على العنصر المقابل:

JSX 
const numbers = [1, 2, 3, 4, 5];

// مضاعفة كل رقم
const doubled = numbers.map(n => n * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// تحويل مصفوفة أرقام لنصوص
const strings = numbers.map(n => `الرقم ${n}`);
// ['الرقم 1', 'الرقم 2', ...]

// مع معامل الفهرس
const indexed = numbers.map((n, i) => `${i+1}. ${n}`);
// ['1. 1', '2. 2', '3. 3', ...]

// تحويل مصفوفة كائنات
const users = [
  { id: 1, name: 'أحمد', age: 25 },
  { id: 2, name: 'سارة', age: 22 },
  { id: 3, name: 'خالد', age: 30 },
];

// استخراج الأسماء فقط
const names = users.map(u => u.name);
// ['أحمد', 'سارة', 'خالد']

// إضافة خاصية جديدة لكل كائن
const usersWithLabel = users.map(u => ({
  ...u,
  label: `${u.name} (${u.age} سنة)`,
  isAdult: u.age >= 18,
}));

// إنشاء عناصر HTML من بيانات
const cards = products.map(p =>
  `<div class="card">
     <h3>${p.name}</h3>
     <p>${p.price} ريال</p>
   </div>`
).join('');
container.innerHTML = cards;

filter() — تصفية العناصر

filter() تُعيد مصفوفة جديدة تحتوي فقط العناصر التي اجتازت الشرط (أعادت الدالة true):

JavaScript 
const products = [
  { id:1, name:'كتاب',      price:45,  category:'تعليم',  inStock:true  },
  { id:2, name:'ساعة',      price:350, category:'إلكترونيات', inStock:false },
  { id:3, name:'قلم',       price:15,  category:'مكتبية', inStock:true  },
  { id:4, name:'هاتف',      price:899, category:'إلكترونيات', inStock:true  },
  { id:5, name:'حقيبة',     price:120, category:'ملابس',  inStock:true  },
  { id:6, name:'سماعات',    price:250, category:'إلكترونيات', inStock:false },
];

// المنتجات المتوفرة فقط
const available = products.filter(p => p.inStock);

// المنتجات بسعر أقل من 200
const cheap = products.filter(p => p.price < 200);

// منتجات الإلكترونيات المتوفرة
const availableElec = products.filter(
  p => p.category === 'إلكترونيات' && p.inStock

// البحث بالاسم
const searchTerm = 'ساعة';
const results = products.filter(p =>
  p.name.includes(searchTerm) ||
  p.category.includes(searchTerm)

// حذف عنصر بالـ id (دون تعديل الأصل)
const withoutId2 = products.filter(p => p.id !== 2);

reduce() — تجميع وتلخيص

reduce() هي الأقوى — تُحوّل المصفوفة إلى قيمة واحدة: مجموع، كائن، مصفوفة أخرى...

JavaScript 
// reduce(callback, initialValue)
// callback(accumulator, currentValue, index, array)

const numbers = [1, 2, 3, 4, 5];

// ══ مجموع الأرقام ══
const sum = numbers.reduce((acc, n) => acc + n, 0);
console.log(sum); // 15

// ══ ضرب الأرقام ══
const product = numbers.reduce((acc, n) => acc * n, 1);
console.log(product); // 120

// ══ الحد الأقصى ══
const max = numbers.reduce((acc, n) => n > acc ? n : acc, -Infinity);
console.log(max); // 5

// ══ مجموع خاصية في مصفوفة كائنات ══
const cart = [
  { name:'كتاب', price:45,  qty:2 },
  { name:'قلم',  price:15,  qty:5 },
  { name:'حقيبة', price:120, qty:1 },
];

const totalPrice = cart.reduce((acc, item) => {
  return acc + (item.price * item.qty);
}, 0);
// 45*2 + 15*5 + 120*1 = 90 + 75 + 120 = 285

// ══ تجميع حسب فئة (Grouping) ══
const grouped = products.reduce((acc, p) => {
  const cat = p.category;
  if (!acc[cat]) acc[cat] = [];
  acc[cat].push(p);
  return acc;
}, {});
// { 'تعليم': [...], 'إلكترونيات': [...], 'ملابس': [...] }

// ══ تحويل مصفوفة كائنات لـ Map بالـ id ══
const usersMap = users.reduce((acc, u) => {
  acc[u.id] = u;
  return acc;
}, {});
// { 1: {id:1,...}, 2: {id:2,...} }
console.log(usersMap[2].name); // 'سارة'

find() وfindIndex()

JavaScript 
const users = [
  { id:1, name:'أحمد',  role:'admin' },
  { id:2, name:'سارة',  role:'user'  },
  { id:3, name:'خالد',  role:'user'  },
  { id:4, name:'نورة',  role:'admin' },
];

// find — يُعيد أول عنصر مطابق (أو undefined)
const admin = users.find(u => u.role === 'admin');
console.log(admin); // { id:1, name:'أحمد', role:'admin' }

const user3 = users.find(u => u.id === 3);
console.log(user3.name); // 'خالد'

const notFound = users.find(u => u.id === 99);
console.log(notFound); // undefined

// findIndex — يُعيد فهرس أول عنصر مطابق (أو -1)
const idx = users.findIndex(u => u.id === 3);
console.log(idx); // 2

// مثال عملي: تحديث عنصر بالـ id
const updateUser = (users, id, updates) => {
  const idx = users.findIndex(u => u.id === id);
  if (idx === -1) return users; // لم يُوجَد
  return [
    ...users.slice(0, idx),
    { ...users[idx], ...updates },
    ...users.slice(idx + 1),
  ];
};

const updated = updateUser(users, 2, { role: 'admin', city: 'جدة' });

some() وevery()

const allFilled = fields.every(f => f.trim().length > 0);

// هل يوجد أي منتج نفد من المخزون؟
const hasOutOfStock = products.some(p => !p.inStock);

2.1

sort() — الترتيب

⚠️ sort() تُعدّل المصفوفة الأصلية — استخدم [...arr].sort() دائماً لنسخها أولاً:

JavaScript 
// ══ ترتيب أبجدي (الافتراضي) ══
const fruits = ['موز', 'تفاح', 'برتقال', 'عنب'];
const sorted = [...fruits].sort(); // نسخة مرتبة
console.log(sorted); // ['برتقال', 'تفاح', 'عنب', 'موز']
console.log(fruits); // لم يتغير ✅

// ⚠️ ترتيب الأرقام بدون دالة مقارنة — خاطئ!
const nums = [10, 2, 100, 5, 20];
console.log([...nums].sort()); // [10, 100, 2, 20, 5] ❌

// ✅ ترتيب الأرقام تصاعدياً
const ascNums = [...nums].sort((a, b) => a - b);
// [2, 5, 10, 20, 100] ✅

// ✅ ترتيب الأرقام تنازلياً
const descNums = [...nums].sort((a, b) => b - a);
// [100, 20, 10, 5, 2] ✅

// ترتيب كائنات حسب خاصية
const employees = [
  { name:'خالد', salary:5000 },
  { name:'أحمد', salary:8000 },
  { name:'سارة', salary:6500 },
];

// ترتيب تصاعدي بالراتب
const byAscSalary = [...employees].sort((a, b) => a.salary - b.salary);

// ترتيب أبجدي بالاسم
const byName = [...employees].sort((a, b) => a.name.localeCompare(b.name, 'ar'));

// ترتيب متعدد: بالراتب ثم بالاسم
const sorted2 = [...employees].sort((a, b) => {
  if (a.salary !== b.salary) return b.salary - a.salary;
  return a.name.localeCompare(b.name, 'ar');
});

الطرق المُعدِّلة للمصفوفة الأصلية (Mutating)

JavaScript 
let arr = [1, 2, 3, 4, 5];

// push — إضافة للنهاية (يُعيد الطول الجديد)
arr.push(6, 7); // [1,2,3,4,5,6,7]

// pop — حذف من النهاية (يُعيد العنصر المحذوف)
const last = arr.pop(); // last=7, arr=[1,2,3,4,5,6]

// unshift — إضافة للبداية
arr.unshift(0); // [0,1,2,3,4,5,6]

// shift — حذف من البداية
const first = arr.shift(); // first=0, arr=[1,2,3,4,5,6]

// splice — إضافة/حذف/استبدال في موضع محدد
// splice(start, deleteCount, ...itemsToAdd)
arr.splice(2, 0, 99);      // إضافة 99 عند الفهرس 2
arr.splice(2, 1);          // حذف العنصر عند الفهرس 2
arr.splice(1, 2, 10, 20);  // استبدال 2 عناصر بـ 10 و20

// reverse — عكس المصفوفة
arr.reverse();              // يعكس الأصل
const rev = [...arr].reverse(); // نسخة معكوسة ✅

// fill — تعبئة بقيمة
new Array(5).fill(0);       // [0,0,0,0,0]
arr.fill(0, 2, 4);          // يملأ من 2 لـ 4 بـ 0

الطرق غير المُعدِّلة (Non-Mutating) ✅

JavaScript 
const arr = [1, 2, 3, 4, 5];

// slice — قطع من المصفوفة (لا تُعدّل الأصل)
arr.slice(1, 3);     // [2, 3] — من 1 إلى 3 (غير شامل)
arr.slice(2);        // [3, 4, 5] — من 2 للنهاية
arr.slice(-2);       // [4, 5] — آخر عنصرين
arr.slice();         // [1,2,3,4,5] — نسخة كاملة

// concat — دمج مصفوفات
const a = [1, 2];
const b = [3, 4];
const c = a.concat(b);         // [1,2,3,4]
const d = [...a, ...b];        // [1,2,3,4] ✅ (Spread أفضل)

// indexOf — الفهرس الأول (أو -1)
arr.indexOf(3);      // 2
arr.indexOf(99);     // -1
arr.lastIndexOf(3);  // آخر موضع

// includes — التحقق من وجود قيمة
arr.includes(3);     // true ✅
arr.includes(99);    // false

// join — تحويل لنص
arr.join(', ');      // '1, 2, 3, 4, 5'
arr.join(' - ');     // '1 - 2 - 3 - 4 - 5'
arr.join('');        // '12345'

// at() — وصول بفهرس سالب
arr.at(-1);          // 5 (الأخير) ✅
arr.at(-2);          // 4

// Array.from() — إنشاء مصفوفة من iterable
Array.from('مرحبا');          // ['م','ر','ح','ب','ا']
Array.from({length:5}, (_,i) => i+1); // [1,2,3,4,5]
Array.from(document.querySelectorAll('li')); // مصفوفة من NodeList

الجزء الثالث: التدمير (Destructuring)

Destructuring يُتيح استخراج قيم من المصفوفات والكائنات بطريقة أنيقة وموجزة.

3.1

تدمير المصفوفات (Array Destructuring)

JavaScript 
const colors = ['أحمر', 'أخضر', 'أزرق', 'أصفر'];

// الطريقة التقليدية
const c1 = colors[0];
const c2 = colors[1];

// ✅ Destructuring
const [first, second, third] = colors;
console.log(first);  // 'أحمر'
console.log(second); // 'أخضر'

// تخطّي عناصر
const [, , blue] = colors; // تخطّي أول وثاني
console.log(blue); // 'أزرق'

// Rest في التدمير
const [primary, ...rest] = colors;
console.log(primary); // 'أحمر'
console.log(rest);    // ['أخضر', 'أزرق', 'أصفر']

// قيمة افتراضية
const [a = 'فارغ', b = 'افتراضي', z = 'لا شيء'] = ['أولى'];
console.log(a); // 'أولى'
console.log(b); // 'افتراضي'
console.log(z); // 'لا شيء'

// ⭐ تبادل المتغيرات بسطر واحد
let x = 1, y = 2;
[x, y] = [y, x];
console.log(x, y); // 2  1

// تدمير قيمة Return من دالة
function getMinMax(arr) {
  return [Math.min(...arr), Math.max(...arr)];
}
const [min, max] = getMinMax([5, 2, 8, 1, 9]);
console.log(min, max); // 1  9

JavaScript 
const user = {

// تدمير متداخل
const person = {

function displayUser({ name, age = 0, role = 'user' }) {

الجزء الرابع: الكائنات المتقدمة (Advanced Objects)

4.1

Object.keys() وObject.values() وObject.entries()

JavaScript 
const product = {
  name:     'هاتف ذكي',
  price:    899,
  category: 'إلكترونيات',
  inStock:  true,
  rating:   4.5,
};

// Object.keys — مصفوفة بأسماء المفاتيح
console.log(Object.keys(product));
// ['name', 'price', 'category', 'inStock', 'rating']

// Object.values — مصفوفة بالقيم
console.log(Object.values(product));
// ['هاتف ذكي', 899, 'إلكترونيات', true, 4.5]

// Object.entries — مصفوفة من أزواج [key, value]
console.log(Object.entries(product));
// [['name','هاتف ذكي'], ['price',899], ...]

// التكرار على خصائص الكائن
Object.entries(product).forEach(([key, value]) => {
  console.log(`${key}: ${value}`);
});

// إيجاد مجموع الأرقام في الكائن
const numericSum = Object.values(product)
  .filter(v => typeof v === 'number')
  .reduce((sum, v) => sum + v, 0);
// 899 + 4.5 = 903.5

// تحويل كائن لمصفوفة للترتيب
const scores = { أحمد: 85, سارة: 92, خالد: 78 };
const sortedByScore = Object.entries(scores)
  .sort(([,a], [,b]) => b - a)
  .map(([name, score]) => ({ name, score }));
// [{ name:'سارة', score:92 }, { name:'أحمد', score:85 }, ...]

دمج ونسخ الكائنات

JavaScript 
const defaults = { color:'أزرق', size:'متوسط', visible:true };
const custom   = { color:'أحمر', font:'Cairo' };

// Object.assign — دمج في الأصل (يُعدّله!)
const merged1 = Object.assign({}, defaults, custom);
// { color:'أحمر', size:'متوسط', visible:true, font:'Cairo' }

Nullish Coalescing (??) وLogical Assignment

JavaScript 
// ?? — يُعيد الجانب الأيمن فقط إذا كان الأيسر null أو undefined
// (مختلف عن || الذي يتجاوب مع كل Falsy)

const name  = null   ?? 'مجهول';    // 'مجهول'
const count = 0      ?? 'لا يوجد';  // 0 ✅ (0 ليس null!)
const text  = ''     ?? 'فارغ';     // '' ✅ (نص فارغ ليس null!)
const score = undefined ?? 50;      // 50

// مقارنة || مع ??
console.log(0   || 'بديل');  // 'بديل' (0 = falsy)
console.log(0   ?? 'بديل');  // 0 (0 ليس null/undefined) ✅
console.log(''  || 'بديل');  // 'بديل'
console.log(''  ?? 'بديل');  // '' ✅

// مثال عملي: إعدادات المستخدم
function getSettings(userConfig = {}) {
  return {
    theme:    userConfig.theme    ?? 'light',
    fontSize: userConfig.fontSize ?? 16,
    lang:     userConfig.lang     ?? 'ar',
    volume:   userConfig.volume   ?? 50,  // 0 يبقى 0 ✅
  };
}

console.log(getSettings({ volume: 0 }));
// { theme:'light', fontSize:16, lang:'ar', volume:0 }

// Logical Assignment Operators
let a = null;
a ??= 'قيمة افتراضية'; // يُسند فقط إذا null/undefined

let b = '';
b ||= 'بديل';           // يُسند إذا falsy

let c = 'موجود';
c &&= c.toUpperCase();  // يُسند إذا truthy

// دالة: ملخص المخزون
function getInventorySummary(products) {
  const active = products.filter(p => p.active);
  const inStock = active.filter(p => p.qty > 0);
  const outOfStock = active.filter(p => p.qty === 0);
  const totalValue = active.reduce((sum, p) => sum + p.price * p.qty, 0);

  return {
    total:      active.length,
    inStock:    inStock.length,
    outOfStock: outOfStock.length,
    totalValue: totalValue,
    avgPrice:   (active.reduce((s, p) => s + p.price, 0) / active.length).toFixed(2),
    topRated:   [...active].sort((a, b) => b.rating - a.rating)[0],
    cheapest:   [...active].sort((a, b) => a.price - b.price)[0],
    priciest:   [...active].sort((a, b) => b.price - a.price)[0],
  };
}

// دالة: تجميع بالفئة
function groupByCategory(products) {
  return products
    .filter(p => p.active)
    .reduce((acc, p) => {
      const { cat, price, rating } = p;
      if (!acc[cat]) acc[cat] = { products:[], totalRevenue:0, avgRating:0 };
      acc[cat].products.push(p);
      acc[cat].totalRevenue += price * p.qty;
      return acc;
    }, {});
}

// دالة: تحليل الطلبات
function analyzeOrders(orders, products) {
  // خريطة المنتجات بالـ id
  const productMap = products.reduce((m, p) => ({ ...m, [p.id]: p }), {});

  // إجمالي الإيرادات
  const revenue = orders.reduce((sum, o) => {
    const product = productMap[o.productId];
    return sum + (product?.price ?? 0) * o.qty;
  }, 0);

  // أكثر عميل شراءً
  const customerOrders = orders.reduce((acc, o) => {
    acc[o.customer] = (acc[o.customer] ?? 0) + o.qty;
    return acc;
  }, {});
  const topCustomer = Object.entries(customerOrders)
    .sort(([,a],[,b]) => b - a)[0];

  // المنتج الأكثر مبيعاً
  const productSales = orders.reduce((acc, o) => {
    acc[o.productId] = (acc[o.productId] ?? 0) + o.qty;
    return acc;
  }, {});
  const [topId] = Object.entries(productSales).sort(([,a],[,b]) => b-a)[0];

  return {
    totalOrders:  orders.length,
    totalRevenue: revenue,
    topCustomer:  { name: topCustomer[0], orders: topCustomer[1] },
    topProduct:   productMap[Number(topId)],
  };
}

// دالة: بحث وتصفية
function searchProducts(products, { query='', cat='', minPrice=0, maxPrice=Infinity, minRating=0 } = {}) {
  return products
    .filter(p => p.active)
    .filter(p => !query || p.name.toLowerCase().includes(query.toLowerCase()))
    .filter(p => !cat  || p.cat === cat)
    .filter(p => p.price >= minPrice && p.price <= maxPrice)
    .filter(p => p.rating >= minRating)
    .sort((a, b) => b.rating - a.rating);
}
