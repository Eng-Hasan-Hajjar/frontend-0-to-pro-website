الجزء الأول: ما هي البرمجة الكائنية (OOP)؟

1.1

التعريف والفلسفة

البرمجة الكائنية (Object-Oriented Programming) هي نمط برمجي يُنظّم الكود حول الكائنات (Objects) بدلاً من الإجراءات والدوال. كل كائن يجمع البيانات (Properties) والوظائف التي تعمل عليها (Methods) في وحدة واحدة متماسكة.

مثال من الواقع: سيارة = كائن له خصائص (اللون، الموديل، السرعة) وطرق (اقلع، أوقف، تسارع، أبطئ).

1.2

المفاهيم الأربعة للـ OOP

🏛️ ركائز البرمجة الكائنية الأربع

1. التغليف (Encapsulation)

← تجميع البيانات والطرق في كائن واحد

← إخفاء التفاصيل الداخلية عن الخارج

← التحكم في الوصول عبر getters/setters

2. الوراثة (Inheritance)

← الكائن الابن يرث خصائص وطرق الأب

← إعادة استخدام الكود بدلاً من التكرار

← تسلسل هرمي: Animal → Dog → GoldenRetriever

3. التعددية (Polymorphism)

← نفس الاسم، سلوك مختلف حسب الكائن

← Animal.speak() → Dog: 'هاو' | Cat: 'مياو'

← Method Overriding في الكلاسات الفرعية

4. التجريد (Abstraction)

← إخفاء التعقيد وإظهار الواجهة البسيطة فقط

← مثل قيادة السيارة بدون معرفة المحرك

تعريف Class بسيط

JavaScript 
// تعريف Class
class Person {
  // constructor يُنفَّذ عند إنشاء كائن جديد
  constructor(name, age, email) {
    this.name  = name;
    this.age   = age;
    this.email = email;
    this.createdAt = new Date().toLocaleDateString('ar-SA');
  }

  // Instance Method — دالة لكل كائن
  greet() {
    return `مرحباً، أنا ${this.name} وعمري ${this.age} سنة`;
  }

  introduce() {
    return `اسمي ${this.name} وبريدي ${this.email}`;
  }

  isAdult() {
    return this.age >= 18;
  }

  // toString — يُستخدم عند تحويل الكائن لنص
  toString() {
    return `Person(${this.name}, ${this.age})`;
  }
}

// إنشاء كائنات (Instances) باستخدام new
const ahmed = new Person('أحمد محمد', 25, 'ahmed@example.com');
const sara  = new Person('سارة علي',  22, 'sara@example.com');

console.log(ahmed.name);        // 'أحمد محمد'
console.log(ahmed.greet());     // 'مرحباً، أنا أحمد محمد وعمري 25 سنة'
console.log(sara.isAdult());    // true
console.log(`${ahmed}`);        // 'Person(أحمد محمد, 25)'

// التحقق من نوع الكائن
console.log(ahmed instanceof Person); // true
console.log(typeof ahmed);           // 'object'

// كل instance مستقل بخصائصه
ahmed.age = 26; // لا يؤثر على sara
console.log(sara.age); // 22 (لم تتغير)

Static Methods والـ Static Properties

JavaScript 
class MathUtils {
  // خاصية static — مشتركة بين جميع الكائنات
  static PI = 3.14159265359;
  static version = '1.0.0';
  static #instanceCount = 0; // private static

  constructor() {
    MathUtils.#instanceCount++;
  }

  // طريقة static — لا تحتاج instance للاستدعاء
  static add(a, b)      { return a + b; }
  static subtract(a, b) { return a - b; }
  static multiply(a, b) { return a * b; }
  static divide(a, b) {
    if (b === 0) throw new Error('لا يمكن القسمة على صفر!');
    return a / b;
  }

  static circleArea(r)   { return this.PI * r * r; }
  static circlePerimeter(r) { return 2 * this.PI * r; }

  static clamp(value, min, max) {
    return Math.min(Math.max(value, min), max);
  }

  static getInstanceCount() { return MathUtils.#instanceCount; }
}

// استخدام static methods بدون إنشاء كائن
console.log(MathUtils.add(10, 5));        // 15
console.log(MathUtils.circleArea(7));     // 153.938...
console.log(MathUtils.PI);                // 3.14159...
console.log(MathUtils.clamp(150, 0, 100)); // 100

// ❌ خطأ — static لا تُستدعى على instance
// const m = new MathUtils();
// m.add(1, 2); // TypeError

// مثال شائع: Factory Method Static
class User {
  constructor(name, role) {
    this.name = name;
    this.role = role;
    this.id   = Date.now();
  }

  // Factory Methods
  static createAdmin(name) {
    return new User(name, 'admin');
  }
  static createGuest() {
    return new User('زائر', 'guest');
  }
  static fromJSON(json) {
    const data = JSON.parse(json);
    return new User(data.name, data.role);
  }
}

const admin = User.createAdmin('أحمد');
const guest = User.createGuest();
console.log(admin.role); // 'admin'

Getters وSetters

JavaScript 
class Temperature {
  #celsius; // Private field

  constructor(celsius) {
    this.#celsius = celsius;
  }

  // Getter — يُستدعى كخاصية (بدون أقواس)
  get celsius()    { return this.#celsius; }
  get fahrenheit() { return this.#celsius * 9/5 + 32; }
  get kelvin()     { return this.#celsius + 273.15; }

  // Setter — يُستدعى عند الإسناد
  set celsius(value) {
    if (value < -273.15) {
      throw new RangeError('لا يمكن أن تكون أقل من الصفر المطلق!');
    }
    this.#celsius = value;
  }

  set fahrenheit(f) {
    this.celsius = (f - 32) * 5/9; // يستخدم الـ setter للتحقق
  }

  toString() {
    return `${this.#celsius}°م (${this.fahrenheit.toFixed(1)}°ف)`;
  }
}

const temp = new Temperature(25);
console.log(temp.celsius);    // 25
console.log(temp.fahrenheit); // 77
console.log(temp.kelvin);     // 298.15

temp.celsius = 100;
console.log(`${temp}`); // '100°م (212.0°ف)'

// temp.celsius = -300; // ❌ RangeError

// مثال OOP متكامل: الشخص مع التحقق
class Person {
  #name;
  #age;

  constructor(name, age) {
    this.name = name; // يمر عبر setter
    this.age  = age;
  }

  get name() { return this.#name; }
  set name(val) {
    if (!val || val.trim().length < 2)
      throw new Error('الاسم يجب أن يكون حرفين على الأقل');
    this.#name = val.trim();
  }

  get age() { return this.#age; }
  set age(val) {
    if (!Number.isInteger(val) || val < 0 || val > 150)
      throw new RangeError('العمر يجب أن يكون بين 0 و150');
    this.#age = val;
  }
}

3.1

Private Fields (#)

الحقول الخاصة (Private Fields) تُعرَّف بإشارة # وتكون مرئية فقط داخل الـ Class — لا يمكن الوصول إليها من الخارج:

JavaScript 
class BankAccount {
  // ══ Private Fields — لا يمكن الوصول من خارج الـ class ══
  #balance;
  #owner;
  #transactionHistory = [];
  #accountNumber;

  constructor(owner, initialBalance = 0) {
    this.#owner         = owner;
    this.#balance       = initialBalance;
    this.#accountNumber = this.#generateAccountNumber();
  }

  // Private Method
  #generateAccountNumber() {
    return 'SA' + Math.random().toString().slice(2, 12).padStart(10, '0');
  }

  #recordTransaction(type, amount, note = '') {
    this.#transactionHistory.push({
      type, amount, note,
      balance: this.#balance,
      date: new Date().toLocaleString('ar-SA'),
    });
  }

  // ══ Public Interface ══
  get balance()      { return this.#balance; }
  get owner()        { return this.#owner; }
  get accountNumber(){ return this.#accountNumber; }

  deposit(amount, note = 'إيداع') {
    if (amount <= 0) throw new Error('المبلغ يجب أن يكون أكبر من صفر');
    this.#balance += amount;
    this.#recordTransaction('إيداع', amount, note);
    return this;  // للـ Method Chaining
  }

  withdraw(amount, note = 'سحب') {
    if (amount <= 0)              throw new Error('المبلغ يجب أن يكون أكبر من صفر');
    if (amount > this.#balance)   throw new Error('رصيد غير كافٍ!');
    this.#balance -= amount;
    this.#recordTransaction('سحب', amount, note);
    return this;
  }

  transfer(amount, targetAccount) {
    this.withdraw(amount, `تحويل إلى ${targetAccount.owner}`);
    targetAccount.deposit(amount, `تحويل من ${this.#owner}`);
    return this;
  }

  getStatement(last = 10) {
    return [...this.#transactionHistory].slice(-last);
  }

  toString() {
    return `حساب ${this.#owner} | الرصيد: ${this.#balance.toLocaleString('ar')} ريال`;
  }
}

// ══ الاستخدام ══
const ahmed = new BankAccount('أحمد محمد', 1000);
const sara  = new BankAccount('سارة علي', 500);

ahmed.deposit(2000, 'راتب شهر يناير')
     .deposit(500, 'مكافأة')
     .withdraw(300, 'فاتورة كهرباء');

ahmed.transfer(800, sara);

console.log(`${ahmed}`); // 'حساب أحمد محمد | الرصيد: 2٬400 ريال'
console.log(`${sara}`);  // 'حساب سارة علي | الرصيد: 1٬300 ريال'

// ❌ لا يمكن الوصول للحقول الخاصة من الخارج
// ahmed.#balance = 1000000; // SyntaxError
// ahmed.#transactionHistory;  // SyntaxError

4.1

الوراثة الأساسية بـ extends وsuper

JavaScript 
// ══ الكلاس الأب (Base Class / Parent Class) ══
class Animal {
  #name;
  #sound;
  #legs;

  constructor(name, sound, legs) {
    this.#name  = name;
    this.#sound = sound;
    this.#legs  = legs;
  }

  get name()  { return this.#name; }
  get sound() { return this.#sound; }
  get legs()  { return this.#legs; }

  speak() {
    return `${this.#name} يقول: ${this.#sound}`;
  }

  move(distance = 10) {
    return `${this.#name} تحرّك ${distance} متر`;
  }

  describe() {
    return `${this.#name} حيوان له ${this.#legs} أرجل`;
  }

  toString() { return `Animal(${this.#name})`; }
}

// ══ كلاس الابن (Derived Class / Child Class) ══
class Dog extends Animal {
  #breed;
  #trained;

  constructor(name, breed, trained = false) {
    // super() يجب أن يُستدعى أولاً قبل استخدام this
    super(name, 'هاو! هاو!', 4);
    this.#breed   = breed;
    this.#trained = trained;
  }

  get breed()   { return this.#breed; }
  get trained() { return this.#trained; }

  // طريقة خاصة بـ Dog
  fetch(item) {
    return `${this.name} جلب الـ ${item}! 🎾`;
  }

  // Method Overriding — تجاوز طريقة الأب
  speak() {
    const base = super.speak(); // استدعاء الأب
    return `${base} (وهو يهز ذيله 🐾)`;
  }

  describe() {
    return `${super.describe()} | السلالة: ${this.#breed} | مُدرَّب: ${this.#trained ? 'نعم' : 'لا'}`;
  }
}

class Cat extends Animal {
  #indoor;

  constructor(name, indoor = true) {
    super(name, 'مياو~', 4);
    this.#indoor = indoor;
  }

  speak() {
    return `${super.speak()} (وتتجاهلك بعدين 😼)`;
  }

  purr() { return `${this.name} تُصدر صوت خرخرة... 😻`; }
}

class Bird extends Animal {
  #canFly;

  constructor(name, canFly = true) {
    super(name, 'تغريد', 2);
    this.#canFly = canFly;
  }

  move(distance = 50) {
    if (this.#canFly) return `${this.name} طار ${distance} متر 🦅`;
    return super.move(distance); // يستخدم طريقة الأب إذا لا يطير
  }
}

// ══ الاستخدام ══
const dog  = new Dog('ماكس', 'جولدن ريتريفر', true);
const cat  = new Cat('ميسة');
const bird = new Bird('عصفور', true);

console.log(dog.speak());     // 'ماكس يقول: هاو! هاو! (وهو يهز ذيله 🐾)'
console.log(cat.speak());     // 'ميسة تقول: مياو~ (وتتجاهلك بعدين 😼)'
console.log(dog.fetch('كرة')); // 'ماكس جلب الـ كرة! 🎾'
console.log(bird.move());     // 'عصفور طار 50 متر 🦅'

// التحقق من الوراثة
console.log(dog instanceof Dog);    // true
console.log(dog instanceof Animal); // true (مهم!)
console.log(cat instanceof Dog);    // false

Prototype Chain — كيف تعمل الوراثة داخلياً

في JavaScript، كل كائن يمتلك مرجعاً لـ Prototype خاص به — وهكذا تعمل الوراثة:

🔗 Prototype Chain
  dog                    → Dog.prototype
  ┌──────────────┐        ┌───────────────────┐
  │ name: 'ماكس' │   ──►  │ fetch()           │
  │ breed: 'جولدن'│        │ speak() (override)│
  │ trained: true │        │ __proto__ ──────────►  Animal.prototype
  └──────────────┘        └───────────────────┘   ┌───────────────────┐
                                                   │ speak()           │
                                                   │ move()            │
                                                   │ describe()        │
                                                   │ __proto__ ──────►  Object.prototype
                                                   └───────────────────┘
  عند dog.speak() — JS يبحث:
  1. في dog نفسه ← لا يوجد
  2. في Dog.prototype ← وجد! يُنفَّذ
  عند dog.move() — JS يبحث:
  1. في dog نفسه ← لا يوجد
  2. في Dog.prototype ← لا يوجد
  3. في Animal.prototype ← وجد! يُنفَّذ
JavaScript 
// التحقق من Prototype Chain
console.log(Object.getPrototypeOf(dog) === Dog.prototype);    // true
console.log(Object.getPrototypeOf(Dog.prototype) === Animal.prototype); // true

// instanceof يتحقق على طول السلسلة كلها
console.log(dog instanceof Dog);    // true
console.log(dog instanceof Animal); // true
console.log(dog instanceof Object); // true (الجميع!)

// الطرق المُعرَّفة على الـ prototype (لا تُكرَّر لكل كائن)
console.log(dog.speak === cat.speak); // false (override مختلف)
console.log(dog.move === cat.move);   // true (نفس طريقة Animal)

// hasOwnProperty — هل الخاصية للكائن نفسه؟
// (لا تنتمي للـ prototype)
console.log(dog.hasOwnProperty('name'));  // true
console.log(dog.hasOwnProperty('speak')); // false (من Dog.prototype)

الجزء الخامس: التعددية (Polymorphism)

5.1

Method Overriding وPolymorphism

Polymorphism يعني «أشكال متعددة» — نفس الواجهة، سلوك مختلف حسب الكائن:

JavaScript 
// ══ نظام الأشكال الهندسية ══
class Shape {
  #color;
  #name;

  constructor(name, color = 'أزرق') {
    this.#name  = name;
    this.#color = color;
  }

  get name()  { return this.#name; }
  get color() { return this.#color; }

  // Polymorphic methods — كل كلاس فرعي يتجاوزها
  area()      { return 0; }
  perimeter() { return 0; }

  describe() {
    return `${this.#name} (${this.#color}) | المساحة: ${this.area().toFixed(2)} | المحيط: ${this.perimeter().toFixed(2)}`;
  }
}

class Circle extends Shape {
  #radius;

  constructor(radius, color) {
    super('دائرة', color);
    this.#radius = radius;
  }

  area()      { return Math.PI * this.#radius ** 2; }
  perimeter() { return 2 * Math.PI * this.#radius; }
  get diameter() { return this.#radius * 2; }
}

class Rectangle extends Shape {
  #width; #height;

  constructor(width, height, color) {
    super('مستطيل', color);
    this.#width  = width;
    this.#height = height;
  }

  area()      { return this.#width * this.#height; }
  perimeter() { return 2 * (this.#width + this.#height); }
  get isSquare() { return this.#width === this.#height; }
}

class Triangle extends Shape {
  #a; #b; #c;

  constructor(a, b, c, color) {
    super('مثلث', color);
    this.#a = a; this.#b = b; this.#c = c;
    const s = this.perimeter() / 2;
    if (s <= 0) throw new Error('أضلاع المثلث غير صالحة');
  }

  perimeter() { return this.#a + this.#b + this.#c; }
  area() {
    const s = this.perimeter() / 2;
    return Math.sqrt(s * (s-this.#a) * (s-this.#b) * (s-this.#c));
  }
}

// ══ Polymorphism في العمل ══
const shapes = [
  new Circle(7, 'أحمر'),
  new Rectangle(10, 5, 'أخضر'),
  new Triangle(3, 4, 5, 'أزرق'),
  new Rectangle(6, 6, 'بنفسجي'),
];

// نفس الكود يعمل مع جميع الأشكال — Polymorphism ✅
shapes.forEach(shape => {
  console.log(shape.describe());
});

// إجمالي المساحات
const totalArea = shapes.reduce((sum, s) => sum + s.area(), 0);
console.log(`المساحة الكلية: ${totalArea.toFixed(2)}`);

// ترتيب حسب المساحة
const sorted = [...shapes].sort((a, b) => b.area() - a.area());
console.log('الأكبر مساحة:', sorted[0].name);

Abstract Class Pattern

JavaScript لا تملك abstract classes رسمياً، لكن يمكن محاكاتها:

JavaScript 
// محاكاة Abstract Class في JavaScript
class AbstractVehicle {
  #make;
  #model;
  #year;

  constructor(make, model, year) {
    // منع إنشاء كائن من Abstract Class مباشرة
    if (new.target === AbstractVehicle) {
      throw new Error('لا يمكن إنشاء كائن من AbstractVehicle مباشرة!');
    }
    this.#make  = make;
    this.#model = model;
    this.#year  = year;
  }

  get make()  { return this.#make; }
  get model() { return this.#model; }
  get year()  { return this.#year; }
  get age()   { return new Date().getFullYear() - this.#year; }

  // Abstract Methods — يجب أن يُطبَّق في الكلاسات الفرعية
  getFuelType() { throw new Error('getFuelType() يجب أن يُنفَّذ!'); }
  start()       { throw new Error('start() يجب أن يُنفَّذ!'); }
  stop()        { throw new Error('stop() يجب أن يُنفَّذ!'); }

  // Concrete Method — موجود في جميع المركبات
  describe() {
    return `${this.#year} ${this.#make} ${this.#model} (${this.getFuelType()})`;
  }
}

class ElectricCar extends AbstractVehicle {
  #batteryLevel;

  constructor(make, model, year, batteryLevel = 100) {
    super(make, model, year);
    this.#batteryLevel = batteryLevel;
  }

  getFuelType() { return 'كهربائي ⚡'; }

  start() {
    if (this.#batteryLevel < 5) return 'البطارية فارغة! اشحن السيارة';
    return `${this.make} ${this.model} تعمل بصمت (${this.#batteryLevel}% بطارية)`;
  }

  stop() { return `${this.make} تتوقف وتُعيد شحن البطارية`; }

  charge(amount) {
    this.#batteryLevel = Math.min(100, this.#batteryLevel + amount);
    return `البطارية: ${this.#batteryLevel}%`;
  }
}

class GasCar extends AbstractVehicle {
  #fuel;

  constructor(make, model, year, fuel = 50) {
    super(make, model, year);
    this.#fuel = fuel;
  }

  getFuelType() { return 'بنزين ⛽'; }
  start()  { return `محرك ${this.make} ${this.model} يبدأ بضجيج (${this.#fuel}L وقود)`; }
  stop()   { return `${this.make} تتوقف`; }
}

// ══ Mixin: وظيفة تُضيف طرقاً لـ class ══

// Mixin للتسلسل الزمني
const TimestampMixin = (Base) => class extends Base {
  #createdAt = new Date();
  #updatedAt = new Date();

  get createdAt() { return this.#createdAt.toLocaleDateString('ar-SA'); }
  get updatedAt() { return this.#updatedAt.toLocaleDateString('ar-SA'); }

  touch() { this.#updatedAt = new Date(); return this; }
};

// Mixin للتحقق
const ValidatableMixin = (Base) => class extends Base {
  #errors = [];

  get errors()  { return [...this.#errors]; }
  get isValid() { return this.#errors.length === 0; }

  addError(field, msg) { this.#errors.push({ field, msg }); }
  clearErrors()        { this.#errors = []; }
};

// Mixin للتسجيل
const LoggableMixin = (Base) => class extends Base {
  #logs = [];

  log(action) {
    this.#logs.push({ action, at: new Date().toLocaleTimeString('ar-SA') });
  }

  getLogs() { return [...this.#logs]; }
};

// تطبيق Mixins متعددة
class BaseProduct {
  constructor(name, price) {
    this.name  = name;
    this.price = price;
  }
}

class Product extends LoggableMixin(ValidatableMixin(TimestampMixin(BaseProduct))) {
  constructor(name, price, stock) {
    super(name, price);
    this.stock = stock;
    this.log('إنشاء منتج');
  }

  validate() {
    this.clearErrors();
    if (!this.name || this.name.length < 2) this.addError('name', 'اسم قصير');
    if (this.price <= 0)  this.addError('price', 'سعر غير صالح');
    if (this.stock < 0)   this.addError('stock', 'كمية سالبة');
    return this.isValid;
  }

  updatePrice(newPrice) {
    const old = this.price;
    this.price = newPrice;
    this.touch();
    this.log(`تغيير السعر: ${old} → ${newPrice}`);
    return this;
  }
}

const product = new Product('هاتف Galaxy', 899, 50);
product.validate();
product.updatePrice(799);
console.log(product.isValid);   // true
console.log(product.createdAt); // تاريخ اليوم
console.log(product.getLogs()); // [{action:'إنشاء منتج',...}, {action:'تغيير السعر',...}]

// ══ 1. Base Class: Person ══
class Person {
  #name; #email; #phone;

  constructor(name, email, phone) {
    this.name  = name;
    this.email = email;
    this.phone = phone;
  }

  get name()  { return this.#name; }
  get email() { return this.#email; }
  get phone() { return this.#phone; }

  set name(val) {
    if (!val?.trim() || val.trim().length < 2)
      throw new Error('الاسم يجب أن يكون حرفين على الأقل');
    this.#name = val.trim();
  }
  set email(val) {
    if (!val?.includes('@')) throw new Error('البريد الإلكتروني غير صالح');
    this.#email = val.toLowerCase().trim();
  }
  set phone(val) {
    if (!/^05[0-9]{8}$/.test(val)) throw new Error('رقم الجوال غير صالح');
    this.#phone = val;
  }

  getContactInfo() {
    return `📧 ${this.#email} | 📞 ${this.#phone}`;
  }

  toString() { return `Person(${this.#name})`; }
}

// ══ 2. Employee Class ══
class Employee extends Person {
  static #totalEmployees = 0;
  static #nextId = 1000;

  #id;
  #department;
  #baseSalary;
  #hireDate;
  #leaveBalance;
  #performanceHistory = [];

  constructor(name, email, phone, department, baseSalary) {
    super(name, email, phone);
    this.#id          = `EMP-${++Employee.#nextId}`;
    this.#department  = department;
    this.#baseSalary  = baseSalary;
    this.#hireDate    = new Date();
    this.#leaveBalance = 21; // 21 يوم إجازة سنوياً
    Employee.#totalEmployees++;
  }

  // Getters
  get id()          { return this.#id; }
  get department()  { return this.#department; }
  get baseSalary()  { return this.#baseSalary; }
  get hireDate()    { return this.#hireDate.toLocaleDateString('ar-SA'); }
  get leaveBalance(){ return this.#leaveBalance; }
  get yearsOfService() {
    return Math.floor((Date.now() - this.#hireDate) / (365.25 * 24 * 60 * 60 * 1000));
  }

  // Polymorphic — كل كلاس يُعرّفه بشكل مختلف
  calculateBonus()  { return 0; }
  calculateSalary() { return this.#baseSalary + this.calculateBonus(); }
  getRole()         { return 'موظف'; }

  addPerformance(score, comment) {
    if (score < 1 || score > 5) throw new RangeError('التقييم بين 1 و5');
    this.#performanceHistory.push({
      score, comment,
      date: new Date().toLocaleDateString('ar-SA'),
    });
  }

  getAvgPerformance() {
    if (this.#performanceHistory.length === 0) return 0;
    const sum = this.#performanceHistory.reduce((s, r) => s + r.score, 0);
    return (sum / this.#performanceHistory.length).toFixed(2);
  }

  requestLeave(days) {
    if (days > this.#leaveBalance) throw new Error('رصيد الإجازة غير كافٍ!');
    this.#leaveBalance -= days;
    return `${this.name}: تمت الموافقة على إجازة ${days} أيام. الرصيد المتبقي: ${this.#leaveBalance}`;
  }

  promote(newDept, salaryIncrease) {
    const old = { dept: this.#department, salary: this.#baseSalary };
    this.#department  = newDept;
    this.#baseSalary += salaryIncrease;
    return `ترقية ${this.name}: ${old.dept} → ${newDept} | الراتب: ${old.salary} → ${this.#baseSalary}`;
  }

  static getTotalEmployees() { return Employee.#totalEmployees; }

  getSummary() {
    return {
      id:          this.#id,
      name:        this.name,
      role:        this.getRole(),
      department:  this.#department,
      salary:      this.calculateSalary(),
      performance: this.getAvgPerformance(),
      hireDate:    this.hireDate,
    };
  }

  toString() {
    return `[${this.#id}] ${this.name} — ${this.getRole()} في ${this.#department}`;
  }
}

// ══ 3. Manager Class ══
class Manager extends Employee {
  #team = [];
  #budget;

  constructor(name, email, phone, department, baseSalary, budget) {
    super(name, email, phone, department, baseSalary);
    this.#budget = budget;
  }

  getRole() { return 'مدير'; }

  // Polymorphism — مكافأة المدير أعلى
  calculateBonus() {
    const avg = Number(this.getAvgPerformance());
    return this.baseSalary * (avg >= 4 ? 0.25 : 0.10);
  }

  addTeamMember(employee) {
    this.#team.push(employee);
    return `${employee.name} انضم لفريق ${this.name}`;
  }

  removeTeamMember(employeeId) {
    const idx = this.#team.findIndex(e => e.id === employeeId);
    if (idx === -1) throw new Error('الموظف غير موجود في الفريق');
    const [removed] = this.#team.splice(idx, 1);
    return `${removed.name} غادر فريق ${this.name}`;
  }

  get teamSize()  { return this.#team.length; }
  get teamNames() { return this.#team.map(e => e.name); }
  get budget()    { return this.#budget; }

  getTeamReport() {
    return {
      manager: this.name,
      teamSize: this.#team.length,
      avgSalary: this.#team.reduce((s, e) => s + e.calculateSalary(), 0) / (this.#team.length || 1),
      avgPerformance: this.#team.reduce((s, e) => s + Number(e.getAvgPerformance()), 0) / (this.#team.length || 1),
    };
  }
}

class Developer extends Employee {
  #techStack;
  #level; // junior, mid, senior

  constructor(name, email, phone, department, baseSalary, techStack, level = 'mid') {
    super(name, email, phone, department, baseSalary);
    this.#techStack = techStack;
    this.#level     = level;
  }

  getRole() { return `مطوّر ${this.#level}`; }

  calculateBonus() {
    const multiplier = { junior: 0.05, mid: 0.10, senior: 0.20 };
    return this.baseSalary * (multiplier[this.#level] ?? 0.10);
  }

  get techStack() { return [...this.#techStack]; }

  addSkill(skill) {
    if (!this.#techStack.includes(skill)) this.#techStack.push(skill);
    return `${this.name} أضاف مهارة: ${skill}`;
  }
}

// ══ 5. HR Department ══
class HRDepartment {
  #employees = new Map();
  #companyName;

  constructor(companyName) {
    this.#companyName = companyName;
  }

  hire(employee) {
    this.#employees.set(employee.id, employee);
    return `تم توظيف ${employee.name} كـ ${employee.getRole()}`;
  }

  fire(employeeId) {
    const emp = this.#employees.get(employeeId);
    if (!emp) throw new Error(`الموظف ${employeeId} غير موجود`);
    this.#employees.delete(employeeId);
    return `تم إنهاء خدمة ${emp.name}`;
  }

  find(id)       { return this.#employees.get(id); }
  getAll()       { return [...this.#employees.values()]; }
  getCount()     { return this.#employees.size; }

  // Polymorphism — يعمل مع جميع أنواع الموظفين
  getPayroll() {
    return this.getAll().map(e => ({
      id:     e.id,
      name:   e.name,
      role:   e.getRole(),
      salary: e.calculateSalary(),
    }));
  }

  getTotalPayroll() {
    return this.getAll().reduce((s, e) => s + e.calculateSalary(), 0);
  }

  getDeptReport() {
    return this.getAll().reduce((acc, emp) => {
      const dept = emp.department;
      if (!acc[dept]) acc[dept] = { count:0, totalSalary:0 };
      acc[dept].count++;
      acc[dept].totalSalary += emp.calculateSalary();
      return acc;
    }, {});
  }

  printReport() {
    console.log(`═`.repeat(60));
    console.log(`   📊 تقرير ${this.#companyName}`);
    console.log(`═`.repeat(60));
    console.log(`   إجمالي الموظفين: ${this.getCount()}`);
    console.log(`   إجمالي الرواتب:  ${this.getTotalPayroll().toLocaleString('ar')} ريال`);
    console.log();
    this.getAll().forEach(e => console.log(`   ${e}`));
  }
}
