الجزء الأول: تصميم الأنواع المشتركة

نبدأ دائماً بالأنواع في مشروع TypeScript جاد — هي "العقد" الذي تتفق عليه كل أجزاء التطبيق (الواجهة، الحالة، API) قبل كتابة أي منطق.

types/task.ts

TypeScript 
export interface Task {
  id: number;
  title: string;
  completed: boolean;
  priority: 'low' | 'medium' | 'high';
  createdAt: string;
}

export type TaskAction =
  | { type: 'SET_TASKS'; payload: Task[] }
  | { type: 'ADD_TASK'; payload: Task }
  | { type: 'TOGGLE_TASK'; payload: { id: number } }
  | { type: 'DELETE_TASK'; payload: { id: number } };

export interface TaskContextValue {
  tasks: Task[];
  dispatch: React.Dispatch<TaskAction>;
}
💡 لاحظ

priority نوع Union محدود ("low" | "medium" | "high")، وTaskAction نوع Discriminated Union — تماماً كما تعلّمنا في الجلستين 25 و30. هذا الملف الواحد سيُستورَد في كل مكوّن وAPI Route لاحقاً.

الجزء الثاني: إدارة الحالة — Context وuseReducer معاً

context/TaskContext.tsx

JSX 
'use client';

import { createContext, useContext, useReducer, ReactNode } from 'react';
import { Task, TaskAction, TaskContextValue } from '../types/task';

const TaskContext = createContext<TaskContextValue | null>(null);

function taskReducer(state: Task[], action: TaskAction): Task[] {
  switch (action.type) {
    case 'SET_TASKS':
      return action.payload;
    case 'ADD_TASK':
      return [...state, action.payload];
    case 'TOGGLE_TASK':
      return state.map((t) =>
        t.id === action.payload.id ? { ...t, completed: !t.completed } : t
      );
    case 'DELETE_TASK':
      return state.filter((t) => t.id !== action.payload.id);
    default:
      return state;
  }
}

export function TaskProvider({ children }: { children: ReactNode }) {
  const [tasks, dispatch] = useReducer(taskReducer, []);
  return (
    <TaskContext.Provider value={{ tasks, dispatch }}>
      {children}
    </TaskContext.Provider>
  );
}

export function useTasks(): TaskContextValue {
  const context = useContext(TaskContext);
  if (!context) throw new Error('useTasks يجب أن يُستخدَم داخل TaskProvider');
  return context;
}
🏛️ هذا بالضبط نمط الجلستين 21 و25 مجتمعتين

Context (21) يحمل الحالة والتحديث معاً بلا Prop Drilling، وuseReducer (25) يُركِّز كل منطق التحديث في مكان واحد — والآن كلاهما مكتوب بأنواع TypeScript دقيقة (30).

الجزء الثالث: API Routes بـ TypeScript

app/api/tasks/route.ts

TypeScript 
import { NextResponse } from 'next/server';
import { Task } from '../../../types/task';

let tasks: Task[] = [
  { id: 1, title: 'تعلّم TypeScript', completed: true, priority: 'high', createdAt: new Date().toISOString() },
  { id: 2, title: 'بناء مشروع التخرّج', completed: false, priority: 'high', createdAt: new Date().toISOString() },
];

export async function GET() {
  return NextResponse.json(tasks);
}

export async function POST(request: Request) {
  const body: { title: string; priority: Task['priority'] } = await request.json();

  const newTask: Task = {
    id: Date.now(),
    title: body.title,
    completed: false,
    priority: body.priority,
    createdAt: new Date().toISOString(),
  };

  tasks.push(newTask);
  return NextResponse.json(newTask, { status: 201 });
}
💡 NextResponse.json

مشابه لـ Response.json من الجلسة 27، لكنه من next/server ويمنح خيارات إضافية (رؤوس مخصَّصة، حالة أسهل). body مكتوب بنوع دقيق مشتق من Task['priority'] — لا any في أي مكان.

الجزء الرابع: صفحة Server Component

app/tasks/page.tsx

TypeScript 
import { Task } from '../../types/task';
import TasksClientView from './TasksClientView';

async function getTasks(): Promise<Task[]> {
  const res = await fetch('http://localhost:3000/api/tasks', { cache: 'no-store' });
  return res.json();
}

export default async function TasksPage() {
  const initialTasks = await getTasks();
  return <TasksClientView initialTasks={initialTasks} />;
}
💡 الحد الفاصل

TasksPage نفسها Server Component (بلا "use client")، تجلب البيانات فقط ثم تُمرِّرها كـ Prop لمكوّن Client Component الذي سيتولّى كل التفاعل. هذا التقسيم هو جوهر الأداء في Next.js.

الجزء الخامس: مكوّنات Client Components التفاعلية

app/tasks/TasksClientView.tsx

JSX 
'use client';

import { useEffect } from 'react';
import { TaskProvider, useTasks } from '../../context/TaskContext';
import { Task } from '../../types/task';
import AddTaskForm from './AddTaskForm';
import TaskItem from './TaskItem';

function TasksInner({ initialTasks }: { initialTasks: Task[] }) {
  const { tasks, dispatch } = useTasks();

  useEffect(() => {
    dispatch({ type: 'SET_TASKS', payload: initialTasks });
  }, [initialTasks, dispatch]);

  return (
    <div>
      <AddTaskForm />
      <ul>
        {tasks.map((task) => (
          <TaskItem key={task.id} task={task} />
        ))}
      </ul>
    </div>
  );
}

export default function TasksClientView({ initialTasks }: { initialTasks: Task[] }) {
  return (
    <TaskProvider>
      <TasksInner initialTasks={initialTasks} />
    </TaskProvider>
  );
}

app/tasks/AddTaskForm.tsx

JSX 
'use client';

import { useState, FormEvent } from 'react';
import { useTasks } from '../../context/TaskContext';
import { Task } from '../../types/task';

export default function AddTaskForm() {
  const [title, setTitle] = useState<string>('');
  const { dispatch } = useTasks();

  async function handleSubmit(e: FormEvent) {
    e.preventDefault();
    if (!title.trim()) return;

    const res = await fetch('/api/tasks', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ title, priority: 'medium' }),
    });
    const newTask: Task = await res.json();

    dispatch({ type: 'ADD_TASK', payload: newTask });
    setTitle('');
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={title} onChange={(e) => setTitle(e.target.value)} placeholder="مهمة جديدة..." />
      <button type="submit">إضافة</button>
    </form>
  );
}

app/tasks/TaskItem.tsx

JSX 
'use client';

import { Task } from '../../types/task';
import { useTasks } from '../../context/TaskContext';

export default function TaskItem({ task }: { task: Task }) {
  const { dispatch } = useTasks();

  return (
    <li className={task.completed ? 'done' : ''}>
      <span onClick={() => dispatch({ type: 'TOGGLE_TASK', payload: { id: task.id } })}>
        {task.title} ({task.priority})
      </span>
      <button onClick={() => dispatch({ type: 'DELETE_TASK', payload: { id: task.id } })}>
        🗑️
      </button>
    </li>
  );
}
