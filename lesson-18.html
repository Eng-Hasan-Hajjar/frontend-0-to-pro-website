React مكتبة JavaScript لبناء واجهات المستخدم باستخدام Virtual DOM

JSX يسمح بكتابة HTML-like syntax داخل JavaScript

المكوّن (Component) دالة JavaScript تُعيد JSX

Props تُمرِّر بيانات من مكوّن أب إلى مكوّن ابن، وهي للقراءة فقط

Component Composition يبني واجهات معقّدة من مكوّنات بسيطة

🎓 أهلاً بمرحلة إدارة الحالة!

في الجلسة الماضية بنينا مكوّنات "ساكنة" — تعرض بيانات تُمرَّر لها عبر Props ولا تتغيّر من تلقاء نفسها.

اليوم نضيف البُعد الناقص: الذاكرة! سنتعلّم كيف يتذكّر المكوّن قيماً تتغيّر بمرور الوقت (نقرة زر، نص مكتوب، عنصر أُضيف) ويُعيد رسم نفسه تلقائياً كلما تغيّرت.

هذا هو useState — أهم Hook ستستخدمه في كل تطبيق React تقريباً.

💡

الجزء الأول: ما هي الحالة (State) ولماذا نحتاجها؟

1.1

المشكلة: متغيرات JavaScript العادية لا تكفي

لنجرّب بناء عدّاد بمتغير JavaScript عادي بدل State، ونرَ ماذا يحدث:

JSX 
function Counter() {
  let count = 0;  // متغير عادي — ليس State!

  function increment() {
    count = count + 1;
    console.log('القيمة الآن:', count); // تتغيّر فعلاً في الذاكرة والكونسول
  }

  return (
    <div>
      <p>العدّاد: {count}</p>
      <button onClick={increment}>زيادة +1</button>
    </div>
  );
}
⚠️ المشكلة

اضغط الزر 10 مرات: ستجد في الـ Console أن count فعلاً وصلت إلى 10.

لكن الرقم المعروض على الشاشة سيبقى 0 دائماً! ❌

السبب: React لا "يراقب" متغيرات JavaScript العادية — تغييرها في الذاكرة لا يُخبر React أن عليه إعادة رسم الواجهة.

1.2

تعريف State

الحالة (State) هي بيانات خاصة بالمكوّن، يُديرها React نفسه، وأي تغيير فيها يُشغِّل تلقائياً إعادة تصيير (Re-render) للمكوّن ليعكس القيمة الجديدة على الشاشة.

  • State = ذاكرة المكوّن — تُحفَظ بين عمليات إعادة الرسم
  • كل تغيير في State عبر الطريقة الصحيحة (setState) يُخبر React: "أعد الرسم"
  • كل مكوّن له State خاص به، منعزل تماماً عن باقي المكوّنات
🎣

الجزء الثاني: useState Hook — أهم Hook في React

2.1

Syntax الأساسي

JSX 
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);
  //     ^ـقيمة الحالة الحالية   ^ـدالة التحديث         ^ـالقيمة الابتدائية

  return (
    <div>
      <p>العدّاد: {count}</p>
      <button onClick={() => setCount(count + 1)}>زيادة +1</button>
    </div>
  );
}
💡 كيف تقرأها

useState(0) يُنشئ متغير حالة بقيمة ابتدائية 0، ويُعيد مصفوفة من عنصرين:

العنصر الأول (count): القيمة الحالية — استخدمها للعرض فقط

العنصر الثاني (setCount): دالة التحديث — الطريقة الوحيدة الصحيحة لتغيير القيمة

الآن: كل ضغطة على الزر تستدعي setCount → React يُحدِّث count ويُعيد رسم الواجهة تلقائياً ✅

2.2

مثال كامل: عدّاد بثلاثة أزرار

JSX 
function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div className="counter">
      <h2>{count}</h2>
      <div className="buttons">
        <button onClick={() => setCount(count - 1)}>➖</button>
        <button onClick={() => setCount(0)}>إعادة تعيين</button>
        <button onClick={() => setCount(count + 1)}>➕</button>
      </div>
    </div>
  );
}
2.3

التحديث الدالي (Functional Updates)

احذر من هذا الخطأ الشائع عند استدعاء setState عدة مرات متتالية:

JavaScript 
// ❌ خطأ شائع
function incrementThreeTimes() {
  setCount(count + 1);
  setCount(count + 1);
  setCount(count + 1);
}
// النتيجة: count تزيد بمقدار 1 فقط، وليس 3!
// السبب: الثلاثة استدعاءات "ترى" نفس القيمة القديمة لـ count (Closure)
JavaScript 
// ✅ الحل: مرّر دالة إلى setCount بدل القيمة مباشرة
function incrementThreeTimesCorrect() {
  setCount(prev => prev + 1);
  setCount(prev => prev + 1);
  setCount(prev => prev + 1);
}
// النتيجة: count تزيد بمقدار 3 فعلاً ✅
// السبب: React يمرّر "أحدث قيمة" لكل دالة تحديث بالترتيب، لا القيمة القديمة المخزَّنة
⭐ القاعدة الذهبية

إن كانت القيمة الجديدة تعتمد على القيمة القديمة، استخدم دائماً الشكل الدالي: setState(prev => ...)

الجزء الثالث: الفرق بين State وProps

المعيارStateProps

3.1

مثال عملي يوضح الفرق

JSX 
function UserCard({ name }) {          // ← name هو Prop: قادم من الخارج، للقراءة فقط
  const [liked, setLiked] = useState(false); // ← liked هو State: يملكه هذا المكوّن فقط

  return (
    <div className="card">
      <h3>{name}</h3>
      <button onClick={() => setLiked(!liked)}>
        {liked ? '❤️ أُعجِبت' : '🤍 إعجاب'}
      </button>
    </div>
  );
}
// name لا يمكن لهذا المكوّن تغييره أبداً — فقط الأب الذي يستدعي <UserCard name="..." />
// liked يملكه هذا المكوّن بالكامل، ويتغيّر بضغطة زر داخلية

الجزء الرابع: State يُعيد تصيير المكوّن (Re-render)

4.1

كيف تعمل إعادة التصيير

JSX 
function Counter() {
  const [count, setCount] = useState(0);
  console.log('🔄 المكوّن يُعاد رسمه، count =', count);

  return <button onClick={() => setCount(count + 1)}>{count}</button>;
}
// كل ضغطة → setCount تُستدعى → React يُنفِّذ الدالة Counter() بالكامل من جديد
// → سطر console.log يُطبَع من جديد بالقيمة المحدَّثة → JSX يُقارَن ويُحدَّث في DOM الحقيقي
4.2

كل نسخة من المكوّن لها State مستقل

JSX 
function App() {
  return (
    <div>
      <Counter />
      <Counter />
      <Counter />
    </div>
  );
}
💡 مهم

الضغط على زر أحد العدّادات الثلاثة لا يؤثر إطلاقاً على البقية — كل <Counter /> يملك نسخته الخاصة تماماً من count، حتى لو كانت نفس الدالة بالضبط.

الجزء الخامس: تحديث State بشكل صحيح — Immutability

React يكتشف تغيّر الحالة بمقارنة "المرجع" (Reference) القديم بالجديد — لذلك يجب دائماً إنشاء نسخة جديدة بدل تعديل الأصل مباشرة.

5.1

تحديث الكائنات (Objects) بأمان

React 
const [user, setUser] = useState({ name: 'سارة', age: 25 });

// ❌ خطأ: تعديل الكائن مباشرة
function wrongUpdate() {
  user.age = 26;
  setUser(user); // نفس المرجع! React قد لا يكتشف أي تغيير
}

// ✅ صحيح: نسخة جديدة بالكامل عبر Spread Operator
function correctUpdate() {
  setUser({ ...user, age: 26 }); // كائن جديد: كل الحقول القديمة + age محدَّث
}
5.2

تحديث المصفوفات (Arrays) بأمان

React 
const [todos, setTodos] = useState([{ id: 1, text: 'تعلّم React' }]);

// ❌ خطأ: push يُعدِّل المصفوفة الأصلية
function addWrong(newTodo) {
  todos.push(newTodo);
  setTodos(todos);
}

// ✅ إضافة عنصر: Spread Operator
function addCorrect(newTodo) {
  setTodos([...todos, newTodo]);
}

// ✅ حذف عنصر: filter بدل splice
function removeTodo(id) {
  setTodos(todos.filter(t => t.id !== id));
}

// ✅ تعديل عنصر واحد: map بدل تعديل مباشر
function toggleTodo(id) {
  setTodos(todos.map(t => t.id === id ? { ...t, done: !t.done } : t));
}
⭐ القاعدة الذهبية

لا تُعدِّل State مباشرة أبداً (لا push، لا splice، لا تغيير خاصية بـ =). أنشئ دائماً نسخة جديدة عبر ...Spread أو map/filter.

الجزء السادس: State في النماذج (Controlled Inputs)

6.1

Controlled Component: الأساسيات

JSX 
function NameInput() {
  const [name, setName] = useState('');

  return (
    <div>
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="اكتب اسمك..."
      />
      <p>مرحباً، {name || 'زائر'}! 👋</p>
    </div>
  );
}
💡 لماذا "Controlled"؟

قيمة input تُقاد بالكامل من state (value={name})، وكل حرف يُكتَب يستدعي onChange فوراً فيُحدِّث name → إعادة رسم → القيمة المعروضة تتزامن تماماً مع ما كتبه المستخدم.

6.2

نموذج متعدد الحقول

JSX 
function RegisterForm() {
  const [form, setForm] = useState({ name: '', email: '', password: '' });

  function handleChange(e) {
    const { name, value } = e.target;
    setForm(prev => ({ ...prev, [name]: value })); // تحديث حقل واحد فقط
  }

  function handleSubmit(e) {
    e.preventDefault();
    console.log('بيانات التسجيل:', form);
  }

  return (
    <form onSubmit={handleSubmit}>
      <input name="name" value={form.name} onChange={handleChange} placeholder="الاسم الكامل" />
      <input name="email" type="email" value={form.email} onChange={handleChange} placeholder="البريد الإلكتروني" />
      <input name="password" type="password" value={form.password} onChange={handleChange} placeholder="كلمة المرور" />
      <button type="submit">تسجيل</button>
    </form>
  );
}

الجزء السابع: رفع الحالة للأعلى (Lifting State Up)

ماذا لو احتاج مكوّنان "أخوان" (Sibling) مشاركة نفس القيمة؟ لا يمكن لمكوّن الوصول مباشرة لحالة مكوّن آخر — الحل: انقل الحالة للأب المشترك الأقرب.

7.1

المشكلة

JSX 
// ❌ كل مكوّن يملك حالته الخاصة المعزولة — لا تواصل بينهما
function TemperatureInput() {
  const [temp, setTemp] = useState('');
  return <input value={temp} onChange={e => setTemp(e.target.value)} />;
}

function TemperatureDisplay() {
  // كيف يصل هذا المكوّن لقيمة temp من المكوّن الآخر؟ مستحيل بهذا الشكل!
  return <p>🌡️ القيمة: ؟</p>;
}
7.2

الحل: رفع الحالة إلى الأب المشترك

JSX 
function TemperatureConverter() {
  const [temp, setTemp] = useState(''); // ← الحالة انتقلت هنا، للأب المشترك

  return (
    <div>
      <TemperatureInput temp={temp} onTempChange={setTemp} />
      <TemperatureDisplay temp={temp} />
    </div>
  );
}

function TemperatureInput({ temp, onTempChange }) {
  return (
    <input value={temp} onChange={(e) => onTempChange(e.target.value)} />
  );
}

function TemperatureDisplay({ temp }) {
  const fahrenheit = temp ? (temp * 9 / 5 + 32).toFixed(1) : '—';
  return <p>🌡️ {temp || 0}° مئوية = {fahrenheit}° فهرنهايت</p>;
}
💡 الفكرة

الحالة الآن تعيش في TemperatureConverter (الأب)، وتُمرَّر للأبناء عبر Props: القيمة نفسها (temp) ودالة تحديثها (onTempChange). أي تغيير في الابن الأول ينعكس فوراً على الابن الثاني، لأن كليهما "يقرأ" من نفس المصدر.

الجزء الثامن: المشروع — قائمة مهام تفاعلية بـ useState

سنبني تطبيق Todo List كامل: إضافة مهمة، تبديل حالة الإكمال، حذف مهمة، وعدّاد للمهام المتبقية — باستخدام useState فقط.

src/App.jsx

JSX 
import { useState } from 'react';

function App() {
  const [todos, setTodos] = useState([
    { id: 1, text: 'تعلّم useState', done: true },
    { id: 2, text: 'بناء تطبيق Todo', done: false },
  ]);
  const [input, setInput] = useState('');

  function addTodo() {
    if (input.trim() === '') return;
    const newTodo = { id: Date.now(), text: input, done: false };
    setTodos([...todos, newTodo]);
    setInput('');
  }

  function toggleTodo(id) {
    setTodos(todos.map(t => (t.id === id ? { ...t, done: !t.done } : t)));
  }

  function deleteTodo(id) {
    setTodos(todos.filter(t => t.id !== id));
  }

  const remaining = todos.filter(t => !t.done).length;

  return (
    <div className="app">
      <h1>📝 قائمة المهام</h1>

      <div className="input-row">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && addTodo()}
          placeholder="مهمة جديدة..."
        />
        <button onClick={addTodo}>إضافة</button>
      </div>

      <ul>
        {todos.map(todo => (
          <li key={todo.id} className={todo.done ? 'done' : ''}>
            <span onClick={() => toggleTodo(todo.id)}>{todo.text}</span>
            <button onClick={() => deleteTodo(todo.id)}>🗑️</button>
          </li>
        ))}
      </ul>

      <p>{remaining} مهمة متبقية من أصل {todos.length}</p>
    </div>
  );
}

export default App;

الجزء التاسع: تمارين الجلسة الثامنة عشرة

✏️ تمرين 1 — عدّاد متقدم15 دقيقة

  • أضف حداً أقصى (لا يتجاوز 10) وحداً أدنى (لا يقل عن 0) لزري + و−
  • أضف زر "×2" يُضاعف القيمة الحالية
  • غيّر لون الرقم إلى أحمر إذا وصل للحد الأقصى أو الأدنى

✏️ تمرين 2 — Toggle وForms20 دقيقة

  • أنشئ مكوّن DarkModeToggle بزر يبدّل بين وضعين ويُغيّر لون خلفية الصفحة
  • أنشئ نموذج بحث بسيط: input يعرض نتيجة فورية أسفله بمجرد الكتابة
  • أضف عداد أحرف تحت حقل نصي (مثال: "12 / 100 حرف")

✏️ تمرين 3 — قائمة تسوّق25 دقيقة

  • مصفوفة منتجات (اسم + سعر) في State
  • زر "إضافة منتج عشوائي" من قائمة جاهزة مسبقاً
  • زر حذف بجانب كل منتج
  • عرض الإجمالي (مجموع الأسعار) في الأسفل، يتحدّث تلقائياً

الواجب: تطبيق ملاحظات لاصقة (Sticky Notes) بـ useState

أنشئ تطبيق React بـ Vite لإدارة ملاحظات لاصقة ملوَّنة — State فقط، بلا أي مكتبات خارجية:

◆ المتطلبات الوظيفية:

  • إضافة ملاحظة جديدة بلون عشوائي من قائمة ألوان محدَّدة مسبقاً
  • حذف ملاحظة بضغطة زر
  • تعديل نص ملاحظة موجودة (تحويلها لحقل نص قابل للتعديل عند الضغط عليها)
  • عداد يعرض "عدد الملاحظات: N" يتحدَّث تلقائياً
  • رسالة "لا توجد ملاحظات بعد ✨" تظهر فقط عندما تكون القائمة فارغة (Conditional Rendering)

◆ متطلبات تقنية:

  • استخدام useState فقط (بلا أي Hook آخر — سنتعلّم الباقي لاحقاً)
  • تحديث المصفوفة بطرق آمنة فقط (map/filter/spread) — بلا push أو splice مباشر
  • كل ملاحظة كائن يحوي: id فريد (Date.now())، النص، اللون
  • تقسيم الكود لمكوّنين على الأقل: App وNoteCard
