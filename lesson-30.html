TypeScript يكتشف أخطاء الأنواع وقت الكتابة، قبل أي تشغيل للكود

الأنواع الأساسية: string وnumber وboolean وarray وtuple

interface يصف شكل كائن؛ ? يجعل حقلاً اختيارياً

تجنّب any قدر الإمكان — unknown أكثر أماناً ويُجبر على التحقّق من النوع أولاً

الأنواع البسيطة تكفي لمتغيّر أو دالة واحدة، لكن ماذا عن دالة تعمل مع أي نوع بيانات، أو حالة قد تكون نصاً أو رقماً؟

اليوم نتعلّم Generics وUnion Types وUtility Types الجاهزة — الأدوات التي تجعل TypeScript مرناً بقدر JavaScript، لكن بأمان كامل.

الجزء الأول: Generics — الأساسيات

1.1

المشكلة: تكرار لكل نوع

TypeScript 
// ❌ بدون Generics: نحتاج دالة منفصلة لكل نوع!
function firstStringElement(arr: string[]): string {
  return arr[0];
}

function firstNumberElement(arr: number[]): number {
  return arr[0];
}
// تكرار! ماذا لو أردنا نفس المنطق بالضبط لأي نوع بيانات؟
1.2

الحل: Generic Function

TypeScript 
// ✅ T يمثّل "أي نوع"، يُحدَّد وقت الاستدعاء
function firstElement<T>(arr: T[]): T {
  return arr[0];
}

firstElement<string>(['أ', 'ب', 'ج']); // T = string
firstElement<number>([1, 2, 3]); // T = number
firstElement(['تلقائي']); // TypeScript يستنتج T = string تلقائياً، بلا كتابتها صراحة
1.3

Generic في الكائنات

TypeScript 
interface Box<T> {
  value: T;
}

const stringBox: Box<string> = { value: 'مرحباً' };
const numberBox: Box<number> = { value: 42 };

function getValue<T>(box: Box<T>): T {
  return box.value;
}

الجزء الثاني: Union Types وIntersection Types

2.1

Union: قيمة من عدة أنواع محتملة

TypeScript 
function printId(id: string | number) {
  console.log(`المعرّف: ${id}`);
}

printId(101); // ✅ صحيح
printId('A101'); // ✅ صحيح أيضاً
printId(true); // ❌ خطأ: boolean غير مسموح في هذا الـ Union
2.2

تضييق النوع (Type Narrowing)

TypeScript 
function formatId(id: string | number): string {
  if (typeof id === 'string') {
    return id.toUpperCase(); // TypeScript يعرف هنا أن id سلسلة نصية بالتأكيد
  }
  return id.toFixed(2); // وهنا يعرف أنه رقم بالتأكيد
}
💡 الفكرة

داخل كل فرع من الشرط، يُضيِّق TypeScript النوع تلقائياً بناءً على فحص typeof — فيسمح باستدعاء طرق خاصة بذلك النوع فقط داخل ذلك الفرع.

2.3

Intersection: دمج عدة أنواع

TypeScript 
interface WithId {
  id: number;
}

interface WithTimestamp {
  createdAt: Date;
}

type Entity = WithId & WithTimestamp; // يجب أن يحقّق الشرطين معاً

const post: Entity = {
  id: 1,
  createdAt: new Date(),
};

الجزء الثالث: Utility Types الجاهزة

3.1

Partial: كل الحقول اختيارية

TypeScript 
interface User {
  id: number;
  name: string;
  email: string;
  age: number;
}

// Partial: مفيد لتحديث جزئي (Update) — لا حاجة لكل الحقول
function updateUser(id: number, changes: Partial<User>) {
  // changes قد تحوي أي مجموعة فرعية من حقول User
}

updateUser(1, { name: 'اسم جديد' }); // ✅ لا حاجة لبقية الحقول
3.2

Pick وOmit

TypeScript 
// Pick: اختيار حقول محدَّدة فقط
type UserPreview = Pick<User, 'id' | 'name'>;
// UserPreview = { id: number; name: string }

// Omit: كل الحقول ما عدا المحدَّدة
type UserWithoutEmail = Omit<User, 'email'>;
// UserWithoutEmail = { id: number; name: string; age: number }
3.3

Record: بناء نوع كائن بمفاتيح وقيم محدَّدة

TypeScript 
type UserRoles = Record<string, 'admin' | 'editor' | 'viewer'>;

const roles: UserRoles = {
  ahmad: 'admin',
  sara: 'editor',

الجزء الرابع: أنواع دقيقة لـ useReducer

4.1

Discriminated Union للأكشنات

TypeScript 
// كل أكشن له type مميّز، وTypeScript يعرف شكل payload المناسب بناءً عليه
type CartAction =
  | { type: 'ADD_ITEM'; payload: { id: number; name: string; price: number } }
  | { type: 'REMOVE_ITEM'; payload: { id: number } }
  | { type: 'CLEAR_CART' };
4.2

reducer مكتوب بأنواع صارمة

TypeScript 
interface CartItem {
  id: number;
  name: string;
  price: number;
  quantity: number;
}

function cartReducer(state: CartItem[], action: CartAction): CartItem[] {
  switch (action.type) {
    case 'ADD_ITEM':
      return [...state, { ...action.payload, quantity: 1 }];
    case 'REMOVE_ITEM':
      return state.filter((item) => item.id !== action.payload.id);
    case 'CLEAR_CART':
      return [];
    default:
      return state;
  }
}
💡 الفائدة الحقيقية

داخل حالة REMOVE_ITEM، لن يسمح TypeScript بالوصول لـ action.payload.name — لأنه غير موجود في ذلك النوع تحديداً. الأخطاء المحتملة تُكتشَف فوراً أثناء الكتابة، لا بعد نشر التطبيق.

الجزء الخامس: Custom Hook بـ Generics

5.1

useLocalStorage بنوع Generic

TypeScript 
import { useState, useEffect } from 'react';

function useLocalStorage<T>(key: string, initialValue: T): [T, (value: T) => void] {
  const [value, setValue] = useState<T>(() => {
    const saved = localStorage.getItem(key);
    return saved !== null ? JSON.parse(saved) : initialValue;
  });

  useEffect(() => {
    localStorage.setItem(key, JSON.stringify(value));
  }, [key, value]);

  return [value, setValue];
}

// الاستخدام: T يُستنتَج تلقائياً من القيمة الابتدائية
const [darkMode, setDarkMode] = useLocalStorage('darkMode', false); // T = boolean

الجزء السادس: المشروع — مكتبة Hooks بـ TypeScript

نبني مكتبة Hooks شخصية كاملة، كل واحد مكتوب بأنواع دقيقة، وuseFetch مستخدماً Generics ليعمل مع أي شكل بيانات.

hooks/useToggle.ts

TypeScript 
import { useState } from 'react';

function useToggle(initialValue: boolean = false): [boolean, () => void] {
  const [value, setValue] = useState<boolean>(initialValue);
  const toggle = () => setValue((prev) => !prev);
  return [value, toggle];
}

export default useToggle;

hooks/useFetch.ts

TypeScript 
import { useState, useEffect } from 'react';

interface FetchState<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
}

function useFetch<T>(url: string): FetchState<T> {
  const [state, setState] = useState<FetchState<T>>({
    data: null,
    loading: true,
    error: null,
  });

  useEffect(() => {
    setState({ data: null, loading: true, error: null });

    fetch(url)
      .then((res) => res.json())
      .then((data: T) => setState({ data, loading: false, error: null }))
      .catch((err) => setState({ data: null, loading: false, error: err.message }));
  }, [url]);

  return state;
}

export default useFetch;

الاستخدام: تحديد T حسب شكل البيانات

JSX 
interface Post {
  id: number;
  title: string;
}

function PostsList() {
  const { data: posts, loading, error } = useFetch<Post[]>('https://api.example.com/posts');

  if (loading) return <p>⏳ جارٍ التحميل...</p>;
  if (error) return <p>❌ {error}</p>;

  return <ul>{posts?.map((p) => <li key={p.id}>{p.title}</li>)}</ul>;
}
