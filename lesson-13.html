Storage Wrapper — مكتبة احترافية

نبني Wrapper يُبسّط التعامل مع localStorage ويُضيف معالجة أخطاء وتاريخ انتهاء صلاحية:

JavaScript 
// ══ Storage Wrapper الاحترافي ══
const storage = {

  // حفظ — يدعم الأنواع المعقدة وtimeToLive
  set(key, value, ttlMinutes = null) {
    try {
      const item = {
        value,
        timestamp: Date.now(),
        expires: ttlMinutes ? Date.now() + ttlMinutes * 60 * 1000 : null,
      };
      localStorage.setItem(key, JSON.stringify(item));
      return true;
    } catch (error) {
      // QuotaExceededError — المساحة ممتلئة
      if (error.name === 'QuotaExceededError') {
        console.warn('localStorage ممتلئ!');
        this.clearExpired(); // حاول إخلاء المساحة
      }
      console.error('فشل الحفظ:', error.message);
      return false;
    }
  },

  // استرجاع — مع التحقق من انتهاء الصلاحية
  get(key, defaultValue = null) {
    try {
      const raw = localStorage.getItem(key);
      if (raw === null) return defaultValue;

      const item = JSON.parse(raw);

      // التحقق من انتهاء الصلاحية
      if (item.expires && Date.now() > item.expires) {
        this.remove(key);
        return defaultValue;
      }

      return item.value ?? defaultValue;
    } catch (error) {
      return defaultValue;
    }
  },

  // حذف
  remove(key) {
    localStorage.removeItem(key);
  },

  // التحقق من الوجود
  has(key) {
    return localStorage.getItem(key) !== null;
  },

  // الحصول على جميع المفاتيح
  keys() {
    return Object.keys(localStorage);
  },

  // حذف المنتهية الصلاحية
  clearExpired() {
    const keys = this.keys();
    let cleared = 0;
    keys.forEach(key => {
      try {
        const item = JSON.parse(localStorage.getItem(key));
        if (item?.expires && Date.now() > item.expires) {
          localStorage.removeItem(key);
          cleared++;
        }
      } catch { /* تجاهل العناصر غير المتوافقة */ }
    });
    console.log(`تم حذف ${cleared} عنصر منتهي الصلاحية`);
  },

  // حجم التخزين المستخدم (تقريبي بالـ KB)
  getSize() {
    let total = 0;
    for (const key of this.keys()) {
      total += (localStorage.getItem(key)?.length ?? 0) + key.length;
    }
    return (total * 2 / 1024).toFixed(2) + ' KB'; // UTF-16
  },

  // مسح الكل
  clear() { localStorage.clear(); },
};

// ══ استخدام الـ Wrapper ══
storage.set('user', { name: 'أحمد', role: 'admin' });
storage.set('token', 'abc123', 60); // ينتهي بعد 60 دقيقة

const user  = storage.get('user', { name: 'زائر' });
const token = storage.get('token');

console.log(storage.has('user'));  // true
console.log(storage.getSize());   // '2.45 KB'

// حفظ بيانات كل خطوة في sessionStorage
class MultiStepForm {
  constructor() {
    this.currentStep = Number(sessionStorage.getItem('formStep') ?? 1);
    this.formData = JSON.parse(sessionStorage.getItem('formData') ?? '{}');
  }

  saveStep(stepData) {
    // دمج بيانات الخطوة الحالية
    this.formData = { ...this.formData, ...stepData };
    sessionStorage.setItem('formData', JSON.stringify(this.formData));
    sessionStorage.setItem('formStep', String(this.currentStep));
  }

  nextStep() {
    this.currentStep++;
    sessionStorage.setItem('formStep', String(this.currentStep));
  }

  prevStep() {
    if (this.currentStep > 1) this.currentStep--;
    sessionStorage.setItem('formStep', String(this.currentStep));
  }

  getAll() { return this.formData; }

  submit() {
    const data = this.getAll();
    console.log('إرسال:', data);
    // بعد الإرسال الناجح:
    sessionStorage.removeItem('formData');
    sessionStorage.removeItem('formStep');
  }
}

// الخطوة الثانية

// إرسال النموذج

JavaScript 
// التعامل مع Cookies يدوياً مؤلم → نبني مساعداً
class CookieManager {

  // إنشاء Cookie
  set(name, value, days = 7, options = {}) {
    const expires = new Date();
    expires.setDate(expires.getDate() + days);

    const cookieParts = [
      `${encodeURIComponent(name)}=${encodeURIComponent(JSON.stringify(value))}`,
      `expires=${expires.toUTCString()}`,
      `path=${options.path ?? '/'}`,
    ];

    if (options.secure) cookieParts.push('Secure');
    if (options.sameSite) cookieParts.push(`SameSite=${options.sameSite}`);

    document.cookie = cookieParts.join('; ');
  }

  // استرجاع Cookie
  get(name) {
    const cookies = document.cookie.split(';');
    for (const cookie of cookies) {
      const [key, val] = cookie.trim().split('=');
      if (decodeURIComponent(key) === name) {
        try {
          return JSON.parse(decodeURIComponent(val));
        } catch {
          return decodeURIComponent(val);
        }
      }
    }
    return null;
  }

  // حذف Cookie
  remove(name) {
    this.set(name, '', -1);
  }

  // التحقق من الوجود
  has(name) {
    return this.get(name) !== null;
  }

  // جميع الـ Cookies
  getAll() {
    return document.cookie.split(';').reduce((acc, c) => {
      const [k, v] = c.trim().split('=');
      if (k) acc[decodeURIComponent(k)] = decodeURIComponent(v ?? '');
      return acc;
    }, {});
  }
}

const cookies = new CookieManager();

cookies.set('user', { name: 'أحمد', role: 'admin' }, 30, { secure: true });
console.log(cookies.get('user'));  // { name: 'أحمد', role: 'admin' }
console.log(cookies.has('user'));  // true
cookies.remove('user');

6.1

نظام تفضيلات المستخدم (User Preferences)

JavaScript 
// نظام تفضيلات شامل
class UserPreferences {
  constructor() {
    this.defaults = {
      theme:    'light',
      language: 'ar',
      fontSize: 16,
      sidebar:  true,
      notifications: true,
    };
    this.prefs = this._load();
  }

  _load() {
    try {
      const saved = localStorage.getItem('userPreferences');
      return saved ? { ...this.defaults, ...JSON.parse(saved) } : { ...this.defaults };
    } catch {
      return { ...this.defaults };
    }
  }

  _save() {
    localStorage.setItem('userPreferences', JSON.stringify(this.prefs));
  }

  get(key) {
    return this.prefs[key] ?? this.defaults[key];
  }

  set(key, value) {
    this.prefs[key] = value;
    this._save();
    this._apply(key, value); // تطبيق فوري
  }

  _apply(key, value) {
    switch (key) {
      case 'theme':
        document.documentElement.setAttribute('data-theme', value);
        break;
      case 'fontSize':
        document.documentElement.style.fontSize = `${value}px`;
        break;
      case 'language':
        document.documentElement.lang = value;
        document.documentElement.dir = value === 'ar' ? 'rtl' : 'ltr';
        break;
    }
  }

  applyAll() {
    Object.entries(this.prefs).forEach(([k, v]) => this._apply(k, v));
  }

  reset() {
    this.prefs = { ...this.defaults };
    this._save();
    this.applyAll();
  }
}

// الاستخدام
const prefs = new UserPreferences();
prefs.applyAll(); // تطبيق عند تحميل الصفحة

// تغيير الثيم
document.querySelector('#theme-toggle').addEventListener('click', () => {
  const newTheme = prefs.get('theme') === 'dark' ? 'light' : 'dark';
  prefs.set('theme', newTheme);
});

class ShoppingCart {
  constructor() {
    this.items = JSON.parse(localStorage.getItem('cart') ?? '[]');
    this._updateBadge();
  }

  _save() {
    localStorage.setItem('cart', JSON.stringify(this.items));

  _updateBadge() {
    const count = this.getTotalQty();
    const badge = document.querySelector('.cart-badge');
    if (badge) {
      badge.textContent = count;
      badge.style.display = count > 0 ? 'block' : 'none';
    }
  }

  add(product) {
    const existing = this.items.find(i => i.id === product.id);
    if (existing) {
      existing.qty++;
    } else {
      this.items.push({ ...product, qty: 1 });
    }
    this._save();
  }

  remove(productId) {
    this.items = this.items.filter(i => i.id !== productId);
    this._save();
  }

  updateQty(productId, qty) {
    if (qty <= 0) { this.remove(productId); return; }
    const item = this.items.find(i => i.id === productId);
    if (item) { item.qty = qty; this._save(); }
  }

  clear() {
    this.items = [];
    this._save();
  }

  getTotalQty()  { return this.items.reduce((s, i) => s + i.qty, 0); }
  getTotalPrice(){ return this.items.reduce((s, i) => s + i.price * i.qty, 0); }
  getCount()     { return this.items.length; }
  isEmpty()      { return this.items.length === 0; }
  getAll()       { return [...this.items]; }
}

// IndexedDB API غير متزامنة وتعتمد على Events
// نستخدم Wrapper مبسط للتعامل معها

class SimpleDB {
  constructor(dbName, version = 1) {
    this.dbName = dbName;
    this.version = version;
    this.db = null;
  }

  // فتح قاعدة البيانات
  open(stores) {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(this.dbName, this.version);

      // إنشاء المخازن عند أول مرة أو Upgrade
      request.onupgradeneeded = (e) => {
        const db = e.target.result;
        stores.forEach(({ name, keyPath, indexes = [] }) => {
          if (!db.objectStoreNames.contains(name)) {
            const store = db.createObjectStore(name, { keyPath });
            indexes.forEach(idx => {
              store.createIndex(idx.name, idx.keyPath, { unique: idx.unique ?? false });
            });
          }
        });
      };

      request.onsuccess = (e) => {
        this.db = e.target.result;
        resolve(this);
      };

      request.onerror = (e) => reject(e.target.error);
    });
  }

  // إضافة/تحديث عنصر
  put(storeName, item) {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction(storeName, 'readwrite');
      const request = tx.objectStore(storeName).put(item);
      request.onsuccess = () => resolve(request.result);
      request.onerror  = () => reject(request.error);
    });
  }

  // استرجاع عنصر
  get(storeName, key) {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction(storeName, 'readonly');
      const request = tx.objectStore(storeName).get(key);
      request.onsuccess = () => resolve(request.result);
      request.onerror  = () => reject(request.error);
    });
  }

  // جميع العناصر
  getAll(storeName) {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction(storeName, 'readonly');
      const request = tx.objectStore(storeName).getAll();
      request.onsuccess = () => resolve(request.result);
      request.onerror  = () => reject(request.error);
    });
  }

  // حذف عنصر
  delete(storeName, key) {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction(storeName, 'readwrite');
      const request = tx.objectStore(storeName).delete(key);
      request.onsuccess = () => resolve();
      request.onerror  = () => reject(request.error);
    });
  }
}

// ══ 1. Storage Module ══
const NotesStorage = {
  KEY: 'notes_app_data',

  getAll() {
    try {
      return JSON.parse(localStorage.getItem(this.KEY) ?? '[]');
    } catch { return []; }
  },

  save(notes) {
    localStorage.setItem(this.KEY, JSON.stringify(notes));
  },

  getSize() {
    const raw = localStorage.getItem(this.KEY) ?? '';
    return ((raw.length * 2) / 1024).toFixed(1) + ' KB';
  }
};

// ══ 2. Notes Manager ══
class NotesApp {
  constructor() {
    this.notes = NotesStorage.getAll();
    this.activeNote = null;
    this.filter = { tag: 'all', search: '' };
    this.autoSaveTimer = null;

    this._bindElements();
    this._bindEvents();
    this._render();
    this._updateStats();
  }

  _bindElements() {
    this.notesList    = document.getElementById('notes-list');
    this.noteEditor   = document.getElementById('note-editor');
    this.titleInput   = document.getElementById('note-title');
    this.contentArea  = document.getElementById('note-content');
    this.tagSelect    = document.getElementById('note-tag');
    this.charCount    = document.getElementById('char-count');
    this.lastSaved    = document.getElementById('last-saved');
    this.notesCount   = document.getElementById('notes-count');
    this.storageSize  = document.getElementById('storage-size');
    this.totalCount   = document.getElementById('total-count');
  }

  _bindEvents() {
    // مذكرة جديدة
    document.getElementById('new-note-btn').addEventListener('click', () => {
      this._openEditor(null);
    });

    // حفظ
    document.getElementById('save-btn').addEventListener('click', () => {
      this._saveNote();
    });

    // حذف
    document.getElementById('delete-btn').addEventListener('click', () => {
      if (this.activeNote && confirm('حذف هذه المذكرة؟')) {
        this._deleteNote(this.activeNote.id);
      }
    });

    // إغلاق المحرر
    document.getElementById('close-editor').addEventListener('click', () => {
      this.noteEditor.classList.add('hidden');
      this.activeNote = null;
    });

    // بحث
    document.getElementById('search-input').addEventListener('input', (e) => {
      this.filter.search = e.target.value.toLowerCase();
      this._render();
    });

    // تصفية بالتصنيف — Event Delegation
    document.querySelector('.tags-filter').addEventListener('click', (e) => {
      const btn = e.target.closest('.tag-btn');
      if (!btn) return;
      document.querySelectorAll('.tag-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      this.filter.tag = btn.dataset.tag;
      this._render();
    });

    // عدّاد الأحرف + حفظ تلقائي
    this.contentArea.addEventListener('input', () => {
      const len = this.contentArea.value.length;
      this.charCount.textContent = `${len} حرف`;
      this._scheduleAutoSave();
    });

    // Event Delegation للمذكرات
    this.notesList.addEventListener('click', (e) => {
      const card = e.target.closest('[data-note-id]');
      if (!card) return;
      const id = Number(card.dataset.noteId);
      const note = this.notes.find(n => n.id === id);
      if (note) this._openEditor(note);
    });
  }

  _openEditor(note) {
    this.activeNote = note;
    this.noteEditor.classList.remove('hidden');

    if (note) {
      this.titleInput.value  = note.title;
      this.contentArea.value = note.content;
      this.tagSelect.value   = note.tag;
    } else {
      this.titleInput.value  = '';
      this.contentArea.value = '';
      this.tagSelect.value   = 'شخصي';
    }

    this.charCount.textContent = `${this.contentArea.value.length} حرف`;
    this.lastSaved.textContent = note ? `آخر حفظ: ${note.updatedAt}` : 'جديدة';
    this.titleInput.focus();
  }

  _saveNote() {
    const title   = this.titleInput.value.trim() || 'بدون عنوان';
    const content = this.contentArea.value.trim();
    const tag     = this.tagSelect.value;
    const now     = new Date().toLocaleString('ar-SA');

    if (this.activeNote) {
      // تحديث موجود
      const idx = this.notes.findIndex(n => n.id === this.activeNote.id);
      this.notes[idx] = { ...this.activeNote, title, content, tag, updatedAt: now };
      this.activeNote  = this.notes[idx];
    } else {
      // مذكرة جديدة
      const newNote = {
        id: Date.now(),
        title, content, tag,
        createdAt: now,
        updatedAt: now,
      };
      this.notes.unshift(newNote);
      this.activeNote = newNote;
    }

    NotesStorage.save(this.notes);
    this.lastSaved.textContent = `محفوظ: ${now}`;
    this._render();
    this._updateStats();
  }

  _deleteNote(id) {
    this.notes = this.notes.filter(n => n.id !== id);
    NotesStorage.save(this.notes);
    this.noteEditor.classList.add('hidden');
    this.activeNote = null;
    this._render();
    this._updateStats();
  }

  _scheduleAutoSave() {
    clearTimeout(this.autoSaveTimer);
    this.autoSaveTimer = setTimeout(() => {
      if (this.activeNote || this.contentArea.value.trim()) {
        this._saveNote();
        this.lastSaved.textContent += ' (تلقائي)';
      }
    }, 2000); // حفظ تلقائي بعد 2 ثانية من التوقف
  }

  _filteredNotes() {
    return this.notes.filter(note => {
      const matchTag    = this.filter.tag === 'all' || note.tag === this.filter.tag;
      const matchSearch = !this.filter.search ||
        note.title.toLowerCase().includes(this.filter.search) ||
        note.content.toLowerCase().includes(this.filter.search);
      return matchTag && matchSearch;
    });
  }

  _render() {
    const notes = this._filteredNotes();

    if (notes.length === 0) {
      this.notesList.innerHTML = `
        <div class="empty-state">
          <p>📭 لا توجد مذكرات</p>
          <p>اضغط '+ مذكرة جديدة' للبدء</p>
        </div>
      `;
      return;
    }

    this.notesList.innerHTML = notes.map(note => `
      <div class="note-card ${this.activeNote?.id === note.id ? 'active' : ''}"
        data-note-id="${note.id}">
        <div class="note-card-header">
          <h3>${note.title}</h3>
          <span class="tag tag--${note.tag}">${note.tag}</span>
        </div>
        <p class="note-preview">${note.content.slice(0, 100)}${note.content.length > 100 ? '...' : ''}</p>
        <small>${note.updatedAt}</small>
      </div>
    `).join('');
  }

  _updateStats() {
    this.notesCount.textContent  = this.notes.length;
    this.totalCount.textContent  = this.notes.length;
    this.storageSize.textContent = NotesStorage.getSize();
  }
}

// ══ تشغيل التطبيق ══
const app = new NotesApp();

// مزامنة بين التبويبات
window.addEventListener('storage', (e) => {
  if (e.key === NotesStorage.KEY) {
    app.notes = NotesStorage.getAll();
    app._render();
    app._updateStats();
  }
});
