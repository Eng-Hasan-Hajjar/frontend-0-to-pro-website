الجزء الثاني: إعداد بيئة React مع Vite

2.1

ما هو Vite؟

Vite هو أداة بناء حديثة وسريعة للغاية. يُقارَن بـ Create React App (CRA) القديم لكنه أسرع منه بعشرات المرات:

المعيارVite vs Create React App
سرعة بدء التشغيلVite: < 1 ثانية | CRA: 30+ ثانية
سرعة Hot ReloadVite: فورية | CRA: ثوانٍ
الحجمVite: خفيف | CRA: ثقيل (مُهمَل رسمياً)
الدعم الرسميReact توصي بـ Vite أو Next.js
دعم الأطرReact, Vue, Svelte, Vanilla JS

الجزء الثالث: JSX — كتابة HTML داخل JavaScript

JSX (JavaScript XML) هي امتداد لـ JavaScript تُتيح كتابة HTML-like code مباشرة في JS. تُحوَّل إلى JavaScript عادي بواسطة Babel/Vite.

3.1

JSX vs HTML — الفروق المهمة

HTMLJSX
class='btn'className='btn' — class كلمة محجوزة في JS
for='name'htmlFor='name' — for كلمة محجوزة في JS
<br><br /> — يجب إغلاق الوسوم الفارغة
<input><input /> — إغلاق ذاتي
onclick='fn()'onClick={fn} — أسلوب camelCase
style='color:red'style={{color: 'red'}} — كائن JS
تعليق HTMLتعليق JSX بـ {/* تعليق */}
عنصران متجاورةيجب wrapper واحد أو Fragment (<></>)

List Rendering والـ key

// القائمة الفارغة
function EmptyAwareList({ items }) {
  if (items.length === 0) return <p>لا توجد عناصر</p>
  return (
    <ul>
      {items.map(item => <li key={item.id}>{item.name}</li>)}
    </ul>
  )
}

الجزء الرابع: المكوّنات (Components)

المكوّن هو اللبنة الأساسية في React — دالة JavaScript تُعيد JSX. فكّر فيه كـ custom HTML element قابل لإعادة الاستخدام.

الجزء الخامس: Props — تمرير البيانات

Props (Properties) هي الطريقة لتمرير البيانات من مكوّن أب لمكوّن ابن. Props للقراءة فقط — لا يمكن تعديلها من الابن.

5.1

تمرير Props واستقبالها

      role='مطوّر Front-End'
      isVerified={true}
      score={95}
      skills={['React', 'JavaScript', 'CSS']}
    />
  )
}

// استقبال Props بـ Destructuring (الأفضل)
function UserCard({ name, role, isVerified = false, score = 0, skills = [] }) {
  //                               قيم افتراضية
  return (
    <div className='user-card'>
      <h2>{name}</h2>
      <p>{role}</p>
      {isVerified && <span>✅ موثّق</span>}
      <p>الدرجة: {score}/100</p>
      {skills.length > 0 && (
        <ul>
          {skills.map((skill, i) => <li key={i}>{skill}</li>)}
        </ul>
      )}
    </div>
  )
}

export default UserCard

الجزء السادس: Component Composition

تركيب المكوّنات هو جوهر React — بناء واجهات معقدة من مكوّنات بسيطة قابلة لإعادة الاستخدام.

6.1

نظام مكوّنات قابلة لإعادة الاستخدام

JSX 
// مكوّن Badge — شارة صغيرة
function Badge({ text, color = 'blue' }) {
  var colorMap = {
    blue:   { bg: '#EBF3FB', text: '#2E75B6' },
    green:  { bg: '#EAF7EE', text: '#1E6B3C' },
    orange: { bg: '#FFF3E0', text: '#B85C00' },
    red:    { bg: '#FFF3F3', text: '#CC0000' },
  }
  var style = colorMap[color] || colorMap.blue

  return (
    <span style={{
      backgroundColor: style.bg,
      color: style.text,
      padding: '2px 10px',
      borderRadius: '20px',
      fontSize: '0.8rem',
      fontWeight: 700,
    }}>
      {text}
    </span>
  )
}

// مكوّن Avatar — صورة دائرية
function Avatar({ src, name, size = 60 }) {
  var fallback = 'https://ui-avatars.com/api/?name=' + encodeURIComponent(name) + '&size=' + size + '&background=2E75B6&color=fff'
  return (
    <img
      src={src || fallback}
      alt={name}
      style={{
        width: size, height: size,
        borderRadius: '50%',
        objectFit: 'cover',
      }}
      onError={function(e){ e.target.src = fallback }}
    />
  )
}

// مكوّن RatingStars
function RatingStars({ rating = 0, max = 5 }) {
  return (
    <div style={{ display: 'flex', gap: '2px' }}>
      {Array.from({ length: max }, function(_, i) {
        return (
          <span key={i} style={{ color: i < rating ? '#E8A020' : '#D0E4F5' }}>★</span>
        )
      })}
    </div>
  )
}

// مكوّن ProductCard يُركّب المكوّنات الأصغر
function ProductCard({ product }) {
  var name=product.name, price=product.price, category=product.category
  var rating=product.rating, inStock=product.inStock, isNew=product.isNew
  return (
    <article className='product-card'>
      <div className='product-card__badges'>
        {isNew && <Badge text='جديد' color='green' />}
        {!inStock && <Badge text='نفد المخزون' color='orange' />}
      </div>
      <Badge text={category} />
      <h3>{name}</h3>
      <RatingStars rating={Math.round(rating)} />
      <p>{price.toLocaleString('ar')} ر.س</p>
      <button disabled={!inStock}>
        {inStock ? 'أضف للسلة' : 'غير متوفر'}
      </button>
    </article>
  )
}

export { Badge, Avatar, RatingStars, ProductCard }

function ProfileCard({ profile }) {
  var name=profile.name, title=profile.title, bio=profile.bio
  var avatar=profile.avatar, location=profile.location
  var skills=profile.skills, badges=profile.badges
  var links=profile.links, stats=profile.stats, available=profile.available

  return (
    <article className='profile-card'>
      <header className='profile-card__header'>
        <div className='profile-card__avatar-wrapper'>
          <Avatar src={avatar} name={name} size={100} />
          {available && <span className='profile-card__available-dot' />}
        </div>
        <div>
          <h2>{name}</h2>
          <p>{title}</p>
          <p>{location}</p>
        </div>
      </header>

      <div className='profile-card__badges'>
        {badges.map(function(b, i) { return <Badge key={i} text={b.text} color={b.color} /> })}
        {available && <Badge text='متاح للعمل' color='green' />}
      </div>

      <p className='profile-card__bio'>{bio}</p>

      <div className='profile-card__skills'>
        <h3>المهارات</h3>
        <div style={{ display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
          {skills.map(function(skill, i) {
            return <span key={i} className='skill-tag'>{skill}</span>
          })}
        </div>
      </div>

      <div className='profile-card__stats'>
        <StatBox value={stats.projects}  label='مشروع' />
        <StatBox value={stats.followers} label='متابع' />
        <StatBox value={stats.stars}     label='نجمة' />
      </div>

      <footer className='profile-card__footer'>
        {links.map(function(link, i) {
          return (
            <a key={i} href={link.url} target='_blank' rel='noreferrer'>
              {link.icon} {link.label}
            </a>
          )
        })}
      </footer>
    </article>
  )
}

export default ProfileCard
