الجزء الأول: لماذا نحتاج البرمجة غير المتزامنة؟

1.1

JavaScript أحادي الخيط (Single-Threaded)

JavaScript ينفّذ تعليمة واحدة في كل مرة — لا يستطيع تنفيذ كودَيْن في آنٍ واحد. هذا يعني أن أي عملية بطيئة (جلب بيانات، قراءة ملف) ستُجمّد الصفحة كلها إذا كانت متزامنة!

🔄 مشكلة الكود المتزامن (Synchronous)
  // تخيّل أن fetchData() تستغرق 3 ثوانٍ
  const data = fetchData();   // ← الصفحة تتجمد 3 ثوانٍ!
  console.log(data);
  updateUI(data);
  // خلال هذه الثلاث ثوانٍ:
  // ❌ لا يمكن النقر على أي زر
  // ❌ لا يمكن التمرير
  // ❌ الأنيميشن يتوقف
  // ❌ المتصفح يُعلم 'الصفحة لا تستجيب'
1.2

Event Loop — كيف يحل JavaScript المشكلة

JavaScript يستخدم Event Loop ومجموعة من الطوابير لتنفيذ العمليات غير المتزامنة دون تجميد الصفحة:

⚙️ كيف يعمل Event Loop
  ┌─────────────────────────────────────────────┐
  │              JavaScript Engine               │
  │                                             │
  │  Call Stack        Web APIs (المتصفح)       │
  │  ┌──────────┐      ┌──────────────────┐     │
  │  │ main()   │  ──► │ setTimeout       │     │
  │  │ func1()  │      │ fetch()          │     │
  │  │ func2()  │      │ addEventListener  │     │
  │  └──────────┘      └────────┬─────────┘     │
  │                             │ انتهت المهمة  │
  │  Event Loop                 ▼               │
  │  ┌──────────────────────────────────────┐   │
  │  │ Callback/Task Queue                  │   │
  │  │ [callback1] [callback2] [callback3]  │   │
  │  └──────────────────────────────────────┘   │
  │  ← Event Loop ينقل للـ Call Stack إذا فارغ  │
  └─────────────────────────────────────────────┘
  1. الكود يُنفَّذ في Call Stack
  2. العمليات الطويلة تُحال للـ Web APIs
  3. عند الانتهاء تُضاف للـ Callback Queue
  4. Event Loop ينقلها للـ Call Stack عندما يفرغ

setTimeout

// عداد تنازلي
function countdown(seconds, onTick, onDone) {
  let remaining = seconds;
  onTick(remaining);

  const timer = setInterval(() => {
    remaining--;
    onTick(remaining);
    if (remaining <= 0) {
      clearInterval(timer);
      onDone();
    }
  }, 1000);

  return timer; // لإمكانية الإلغاء من الخارج
}

الجزء الثالث: Callbacks ومشكلة Callback Hell

3.1

ما هو Callback؟

Callback هي دالة تُمرَّر كمعامل لدالة أخرى لتُنفَّذ لاحقاً (عند انتهاء عملية ما):

JavaScript 
// Callback بسيط
function doWork(task, onDone) {
  console.log(`أبدأ: ${task}`);
  setTimeout(() => {
    const result = `نتيجة: ${task}`;
    onDone(result); // استدعاء الـ callback عند الانتهاء
  }, 1000);
}

doWork('تحميل البيانات', (result) => {
  console.log(result); // 'نتيجة: تحميل البيانات'
});

// Callbacks في أمثلة حقيقية
// addEventListener — يأخذ callback
btn.addEventListener('click', () => console.log('نُقر!'));

// Array methods — تأخذ callback
[1,2,3].forEach(n => console.log(n));

// setTimeout — يأخذ callback
setTimeout(() => console.log('بعد ثانية'), 1000);
3.2

مشكلة Callback Hell (جحيم الـ Callbacks)

عند تداخل العمليات غير المتزامنة، يتحول الكود لهرم من الـ Callbacks يصعب قراءته وصيانته:

JavaScript 
// ❌ Callback Hell — صعب القراءة والصيانة
getUser(userId, (user) => {
  if (user.error) {
    handleError(user.error);
  } else {
    getUserPosts(user.id, (posts) => {
      if (posts.error) {
        handleError(posts.error);
      } else {
        getPostComments(posts[0].id, (comments) => {
          if (comments.error) {
            handleError(comments.error);
          } else {
            getLikes(comments[0].id, (likes) => {
              // هرم يميل لليمين — 'Pyramid of Doom'
              console.log(likes);
            });
          }
        });
      }
    });
  }
});

// المشاكل:
// 1. قراءة صعبة ← تداخل عميق
// 2. معالجة الأخطاء متكررة
// 3. صيانة كابوسية
// 4. لا يمكن استخدام try/catch
// الحل: Promises ثم async/await ✅

الجزء الرابع: الوعود (Promises)

Promise هو كائن يمثّل نتيجة عملية غير متزامنة ستكتمل في المستقبل. مثل: طلب طعام من المطعم — تستلم رقم الطلب (Promise) وتواصل حياتك حتى يُنادى عليك.

4.1

حالات Promise الثلاث

🔄 حالات Promise
  ┌─────────────────────────────────────────────────┐
  │                                                 │
  │   Pending (معلّقة)                              │
  │   الحالة الابتدائية — العملية قيد التنفيذ       │
  │                    │                            │
  │         ┌──────────┴──────────┐                 │
  │         ▼                     ▼                 │
  │   Fulfilled (منجزة)    Rejected (مرفوضة)        │
  │   العملية نجحت         العملية فشلت              │
  │   → then() يُستدعى     → catch() يُستدعى        │
  │                                                 │
  │   ملاحظة: Settled = Fulfilled OR Rejected       │
  └─────────────────────────────────────────────────┘
4.2

إنشاء Promise

JavaScript 
// new Promise((resolve, reject) => { ... })

// Promise بسيطة
const myPromise = new Promise((resolve, reject) => {
  const success = true;

  setTimeout(() => {
    if (success) {
      resolve('تمت العملية بنجاح! ✅'); // Fulfilled
    } else {
      reject(new Error('فشلت العملية ❌')); // Rejected
    }
  }, 2000);
});

// محاكاة جلب بيانات مستخدم
function fetchUser(id) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      const users = {
        1: { id:1, name:'أحمد', email:'ahmed@ex.com' },
        2: { id:2, name:'سارة', email:'sara@ex.com'  },
      };

      const user = users[id];
      if (user) {
        resolve(user);
      } else {
        reject(new Error(`المستخدم ${id} غير موجود`));
      }
    }, 1000);
  });
}

// محاكاة جلب منشورات
function fetchPosts(userId) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (userId) {
        resolve([
          { id:1, title:'منشور أول', userId },
          { id:2, title:'منشور ثاني', userId },
        ]);
      } else {
        reject(new Error('userId مطلوب'));
      }
    }, 800);
  });
}
4.3

then / catch / finally

JavaScript 
// ══ الاستخدام الأساسي ══
fetchUser(1)
  .then(user => {
    console.log('المستخدم:', user.name);
    return user; // تمرير للـ then التالي
  })
  .catch(error => {
    console.error('خطأ:', error.message);
  })
  .finally(() => {
    console.log('انتهت العملية (نجاح أو فشل)');
    hideLoader(); // إخفاء مؤشر التحميل دائماً
  });

// ══ Promise Chaining — البديل عن Callback Hell ✅ ══
fetchUser(1)
  .then(user => {
    console.log('المستخدم:', user.name);
    return fetchPosts(user.id); // يُعيد Promise جديدة
  })
  .then(posts => {
    console.log('المنشورات:', posts.length);
    return fetchComments(posts[0].id);
  })
  .then(comments => {
    console.log('التعليقات:', comments.length);
  })
  .catch(error => {
    // catch واحد يصطاد أخطاء جميع الـ then السابقة
    console.error('خطأ في أي مرحلة:', error.message);
  })
  .finally(() => {
    hideLoader();
  });

// ══ Promise.resolve() وPromise.reject() ══
// Promise جاهزة النتيجة فوراً
Promise.resolve('قيمة فورية')
  .then(val => console.log(val)); // 'قيمة فورية'

Promise.reject(new Error('خطأ فوري'))
  .catch(err => console.error(err.message));
4.4

Promise المتقدمة: all وrace وallSettled وany

JavaScript 
const p1 = fetchUser(1);    // 1 ثانية
const p2 = fetchPosts(1);   // 0.8 ثانية
const p3 = fetchComments(1); // 1.2 ثانية

// ══ Promise.all — انتظر الجميع (يفشل إذا فشل أي منها) ══
Promise.all([p1, p2, p3])
  .then(([user, posts, comments]) => {
    // ينتظر أطول واحد (1.2 ثانية)
    console.log(user.name, posts.length, comments.length);
  })
  .catch(err => {
    // إذا فشلت أي Promise → catch يُستدعى فوراً
    console.error('فشل أحدها:', err.message);
  });

// ══ Promise.race — أول من ينتهي يفوز ══
const timeout = new Promise((_, reject) => {
  setTimeout(() => reject(new Error('انتهت المهلة')), 5000);
});

Promise.race([fetchUser(1), timeout])
  .then(user => console.log('وصلنا في الوقت:', user.name))
  .catch(err => console.error('تأخر:', err.message));

// ══ Promise.allSettled — انتظر الجميع بغض النظر ✅ ══
// لا يفشل — يُعيد نتيجة كل Promise
Promise.allSettled([p1, p2, Promise.reject(new Error('خطأ'))])
  .then(results => {
    results.forEach((result, i) => {
      if (result.status === 'fulfilled') {
        console.log(`P${i+1} نجح:`, result.value);
      } else {
        console.log(`P${i+1} فشل:`, result.reason.message);
      }
    });
  });

// ══ Promise.any — أول من ينجح يفوز ══
// يفشل فقط إذا فشلت الجميع (AggregateError)
Promise.any([
  Promise.reject(new Error('فشل 1')),
  fetchUser(1),
  Promise.reject(new Error('فشل 2')),
])
  .then(first => console.log('أول من نجح:', first.name));
الطريقةالسلوك والاستخدام
Promise.all([...])ينتظر الجميع | يفشل إذا فشل أي — لطلبات متوازية مترابطة
Promise.race([...])أول من ينتهي (نجاح أو فشل) — للـ timeout
Promise.allSettled([...])ينتظر الجميع بغض النظر ✅ — للتقارير الشاملة
Promise.any([...])أول من ينجح — للبحث في مصادر متعددة

الجزء الخامس: async / await

async/await هي صياغة حديثة تجعل كتابة كود غير متزامن يبدو ويُقرأ كأنه كود متزامن عادي — أنيق، واضح، وسهل التصحيح.

5.1

الأساسيات

React 
// async — تجعل الدالة تُعيد Promise دائماً
async function sayHello() {
  return 'مرحباً!';
}
// مكافئ لـ: function sayHello() { return Promise.resolve('مرحباً!'); }

sayHello().then(msg => console.log(msg)); // 'مرحباً!'

// await — ينتظر حتى تنتهي Promise (داخل async فقط)
async function getUser() {
  console.log('جار الجلب...');
  const user = await fetchUser(1);  // ينتظر هنا
  console.log('المستخدم:', user.name); // يُنفَّذ بعد الانتهاء
  return user;
}

getUser().then(user => console.log('تم:', user));

// مقارنة: Promise Chaining مقابل async/await
// ══ بـ Promises ══
function getDataWithPromises() {
  return fetchUser(1)
    .then(user => fetchPosts(user.id))
    .then(posts => posts[0])
    .then(post => fetchComments(post.id));
}

// ══ بـ async/await ✅ ══
async function getDataWithAsync() {
  const user    = await fetchUser(1);
  const posts   = await fetchPosts(user.id);
  const post    = posts[0];
  const comments = await fetchComments(post.id);
  return comments;
}

// Arrow async function
const loadProfile = async (id) => {
  const user = await fetchUser(id);
  return user;
};

معالجة الأخطاء مع try/catch

React 
// try/catch مع async/await — نفس مبدأ الكود المتزامن
async function loadUserProfile(userId) {
  try {
    // كل هذا داخل try
    const user     = await fetchUser(userId);
    const posts    = await fetchPosts(user.id);
    const comments = await fetchComments(posts[0]?.id);

    return { user, posts, comments };

  } catch (error) {
    // يصطاد أي خطأ في أي await أعلاه
    console.error('خطأ في تحميل البروفايل:', error.message);
    return null;

  } finally {
    // يُنفَّذ دائماً (نجاح أو فشل)
    hideLoadingSpinner();
  }
}

// معالجة أخطاء منفصلة لكل await
async function robustLoader() {
  let user, posts;

  try {
    user = await fetchUser(1);
  } catch (e) {
    console.error('فشل جلب المستخدم:', e.message);
    return; // إيقاف التنفيذ
  }

  try {
    posts = await fetchPosts(user.id);
  } catch (e) {
    // فشل جلب المنشورات لكن المستخدم جُلب بنجاح
    posts = []; // قيمة افتراضية
    console.warn('تعذّر جلب المنشورات');
  }

  return { user, posts };
}

// نمط: async مع .catch() خارجي
loadUserProfile(1)
  .then(profile => displayProfile(profile))
  .catch(err => showErrorMessage(err.message));

الجزء السادس: Fetch API — جلب البيانات من الإنترنت

Fetch API هي الطريقة الحديثة المُدمَجة في المتصفح لإرسال طلبات HTTP واستقبال البيانات من الخوادم والـ APIs.

6.1

الطلب الأساسي GET

React 
// fetch(url) — يُعيد Promise
// يحتاج خطوتين: أولاً response ثم البيانات

// ══ مع .then() ══
fetch('https://jsonplaceholder.typicode.com/users/1')
  .then(response => {
    console.log(response.status);   // 200
    console.log(response.ok);       // true (200-299)
    return response.json();         // تحويل JSON لـ Object
  })
  .then(user => {
    console.log(user.name);
  })
  .catch(err => {
    console.error('خطأ في الاتصال:', err.message);
  });

// ══ مع async/await ✅ ══
async function getUser(id) {
  try {
    const response = await fetch(
      `https://jsonplaceholder.typicode.com/users/${id}`
    );

    // التحقق من نجاح الطلب
    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }

    const user = await response.json();
    return user;

  } catch (error) {
    console.error('فشل جلب المستخدم:', error.message);
    throw error; // إعادة رمي للـ caller
  }
}

معالجة أخطاء Fetch بشكل شامل

React 
// Fetch لا يُلقي خطأً لأكواد HTTP الخاطئة (404, 500)!
// يُلقي خطأً فقط لمشاكل الشبكة (لا إنترنت، CORS)

// دالة fetch شاملة مع معالجة أخطاء كاملة
async function apiFetch(url, options = {}) {
  try {
    const response = await fetch(url, {
      headers: { 'Content-Type': 'application/json' },
      ...options,
    });

    // التحقق من HTTP Status
    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(
        errorData.message || `HTTP ${response.status}: ${response.statusText}`
      );
    }

    // التحقق من نوع المحتوى
    const contentType = response.headers.get('content-type');
    if (contentType?.includes('application/json')) {
      return response.json();
    }
    return response.text();

  } catch (error) {
    // مشكلة شبكة (لا إنترنت، CORS)
    if (error.name === 'TypeError') {
      throw new Error('تحقق من اتصالك بالإنترنت');
    }
    throw error;
  }
}

// مثال استخدام مع timeout
async function fetchWithTimeout(url, ms = 5000) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), ms);

  try {
    const res = await fetch(url, { signal: controller.signal });
    clearTimeout(timeoutId);
    return res.json();
  } catch (error) {
    if (error.name === 'AbortError') {
      throw new Error('انتهت مهلة الطلب');
    }
    throw error;
  }
}

// تحتاج مفتاح API مجاني من openweathermap.org
const API_KEY = 'YOUR_FREE_API_KEY';

async function getWeather(city) {
  try {
    const url = `https://api.openweathermap.org/data/2.5/weather`,
      + `?q=${encodeURIComponent(city)}`
      + `&appid=${API_KEY}`
      + `&units=metric`
      + `&lang=ar`;

    const response = await fetch(url);

    if (!response.ok) {
      if (response.status === 404) throw new Error('المدينة غير موجودة');
      throw new Error(`خطأ ${response.status}`);
    }

    const data = await response.json();

    return {
      city:        data.name,
      country:     data.sys.country,
      temp:        Math.round(data.main.temp),
      feelsLike:   Math.round(data.main.feels_like),
      humidity:    data.main.humidity,
      description: data.weather[0].description,
      icon:        data.weather[0].icon,
      wind:        data.wind.speed,
    };

  } catch (error) {
    throw new Error(`تعذّر جلب الطقس: ${error.message}`);
  }
}
